
# Script 2: PlayerMovement.cs - Movimento fiel ao TOP (clique para mover)

player_movement = '''using UnityEngine;
using UnityEngine.AI;
using Mirror;
using UnityEngine.EventSystems;

/// <summary>
/// Sistema de movimento do jogador baseado em Tales of Pirates
/// Click-to-move com NavMeshAgent
/// </summary>
[RequireComponent(typeof(NavMeshAgent))]
[RequireComponent(typeof(Animator))]
[RequireComponent(typeof(CharacterStats))]
public class PlayerMovement : NetworkBehaviour
{
    #region Configuração
    [Header("Movimento")]
    [SerializeField] private float baseMoveSpeed = 5f;
    [SerializeField] private float runSpeedMultiplier = 1.5f;
    [SerializeField] private float rotationSpeed = 10f;
    [SerializeField] private float stoppingDistance = 0.2f;
    
    [Header("Layers")]
    [SerializeField] private LayerMask groundLayer = 1; // Default
    [SerializeField] private LayerMask interactableLayer;
    [SerializeField] private LayerMask enemyLayer;
    
    [Header("Animação")]
    [SerializeField] private string moveSpeedParam = "MoveSpeed";
    [SerializeField] private string isMovingParam = "IsMoving";
    [SerializeField] private string idleParam = "IdleType";
    
    [Header("Configurações TOP")]
    [SerializeField] private bool autoRunToTarget = true; // Correr até alvo ao atacar
    [SerializeField] private float interactionRange = 3f;
    #endregion

    #region Componentes
    private NavMeshAgent agent;
    private Animator anim;
    private CharacterStats stats;
    private Camera mainCamera;
    #endregion

    #region Estado
    private bool isMovementEnabled = true;
    private bool isRunning = false;
    private Vector3 lastClickPosition;
    private Transform currentTarget;
    private float currentSpeed;
    private float targetSpeed;
    #endregion

    #region Eventos
    public event System.Action OnMovementStarted;
    public event System.Action OnMovementStopped;
    public event System.Action<Vector3> OnDestinationSet;
    public event System.Action<IInteractable> OnTargetInteractable;
    #endregion

    #region Propriedades
    public bool IsMoving => agent.hasPath && agent.remainingDistance > agent.stoppingDistance + 0.1f;
    public Vector3 CurrentDestination => agent.hasPath ? agent.destination : transform.position;
    public float CurrentSpeed => currentSpeed;
    public bool IsRunning => isRunning;
    #endregion

    void Awake()
    {
        agent = GetComponent<NavMeshAgent>();
        anim = GetComponent<Animator>();
        stats = GetComponent<CharacterStats>();
        mainCamera = Camera.main;
        
        if (agent == null)
        {
            Debug.LogError("[PlayerMovement] NavMeshAgent não encontrado!", this);
            enabled = false;
            return;
        }
        
        ConfigureAgent();
    }

    void Start()
    {
        if (!isLocalPlayer)
        {
            enabled = false;
            return;
        }
        
        if (mainCamera == null)
        {
            mainCamera = Camera.main;
        }
    }

    void Update()
    {
        if (!isLocalPlayer || !isMovementEnabled) return;
        
        HandleInput();
        UpdateMovementState();
        UpdateAnimation();
    }

    #region Configuração
    private void ConfigureAgent()
    {
        agent.speed = baseMoveSpeed;
        agent.stoppingDistance = stoppingDistance;
        agent.acceleration = baseMoveSpeed * 3f;
        agent.angularSpeed = rotationSpeed * 100f;
        agent.autoBraking = true;
        agent.autoRepath = true;
        agent.obstacleAvoidanceType = ObstacleAvoidanceType.HighQualityObstacleAvoidance;
    }

    public void UpdateMoveSpeed(float newSpeed)
    {
        baseMoveSpeed = newSpeed;
        if (agent != null)
        {
            agent.speed = isRunning ? baseMoveSpeed * runSpeedMultiplier : baseMoveSpeed;
        }
    }
    #endregion

    #region Input
    private void HandleInput()
    {
        // Ignora clique se estiver sobre UI
        if (EventSystem.current != null && EventSystem.current.IsPointerOverGameObject())
            return;
        
        // Clique esquerdo - Mover ou Interagir
        if (Input.GetMouseButtonDown(0))
        {
            HandleLeftClick();
        }
        
        // Clique direito - Atacar ou Interação secundária
        if (Input.GetMouseButtonDown(1))
        {
            HandleRightClick();
        }
        
        // Tecla Shift - Correr
        isRunning = Input.GetKey(KeyCode.LeftShift) || Input.GetKey(KeyCode.RightShift);
        
        // Tecla de parar (S ou Escape no TOP)
        if (Input.GetKeyDown(KeyCode.S) || Input.GetKeyDown(KeyCode.Escape))
        {
            StopMovement();
        }
    }

    private void HandleLeftClick()
    {
        Ray ray = mainCamera.ScreenPointToRay(Input.mousePosition);
        
        // Verifica interagível primeiro
        if (Physics.Raycast(ray, out RaycastHit hit, Mathf.Infinity, interactableLayer))
        {
            if (hit.collider.TryGetComponent<IInteractable>(out var interactable))
            {
                float distance = Vector3.Distance(transform.position, hit.point);
                
                if (distance <= interactionRange)
                {
                    // Interage diretamente
                    interactable.OnInteract(this);
                    OnTargetInteractable?.Invoke(interactable);
                }
                else
                {
                    // Move até o interagível e depois interage
                    MoveToInteract(interactable, hit.point);
                }
                return;
            }
        }
        
        // Move para o chão
        if (Physics.Raycast(ray, out RaycastHit groundHit, Mathf.Infinity, groundLayer))
        {
            SetDestination(groundHit.point);
            lastClickPosition = groundHit.point;
            
            // Efeito visual de clique (opcional)
            ShowClickEffect(groundHit.point);
        }
    }

    private void HandleRightClick()
    {
        Ray ray = mainCamera.ScreenPointToRay(Input.mousePosition);
        
        // Verifica inimigo
        if (Physics.Raycast(ray, out RaycastHit hit, Mathf.Infinity, enemyLayer))
        {
            if (hit.collider.TryGetComponent<IDamageable>(out var damageable))
            {
                // Notifica sistema de combate
                GetComponent<PlayerCombat>()?.SetTarget(hit.transform);
                
                // Move até o inimigo se necessário
                if (autoRunToTarget)
                {
                    float attackRange = GetComponent<PlayerCombat>()?.AttackRange ?? 2f;
                    MoveToTarget(hit.transform, attackRange * 0.8f);
                }
                return;
            }
        }
        
        // Se não clicou em inimigo, comportamento igual ao esquerdo
        HandleLeftClick();
    }
    #endregion

    #region Movimento
    [Client]
    public void SetDestination(Vector3 destination)
    {
        if (!NavMesh.SamplePosition(destination, out NavMeshHit navHit, 5f, NavMesh.AllAreas))
        {
            Debug.LogWarning("[PlayerMovement] Destino fora do NavMesh!");
            return;
        }
        
        agent.SetDestination(navHit.position);
        currentTarget = null;
        
        OnDestinationSet?.Invoke(navHit.position);
        
        // Sincroniza com servidor
        CmdSetDestination(navHit.position);
    }

    [Command]
    private void CmdSetDestination(Vector3 destination)
    {
        if (NavMesh.SamplePosition(destination, out NavMeshHit navHit, 5f, NavMesh.AllAreas))
        {
            agent.SetDestination(navHit.position);
        }
    }

    [Client]
    public void MoveToTarget(Transform target, float stopDistance)
    {
        if (target == null) return;
        
        currentTarget = target;
        agent.stoppingDistance = stopDistance;
        
        // Atualiza destino continuamente enquanto o alvo se move
        StartCoroutine(FollowTarget(target, stopDistance));
    }

    private System.Collections.IEnumerator FollowTarget(Transform target, float stopDistance)
    {
        while (currentTarget == target && target != null)
        {
            if (Vector3.Distance(transform.position, target.position) > stopDistance)
            {
                agent.SetDestination(target.position);
            }
            else
            {
                agent.ResetPath();
                break;
            }
            
            yield return new WaitForSeconds(0.1f); // Atualiza a cada 100ms
        }
        
        agent.stoppingDistance = stoppingDistance;
    }

    private void MoveToInteract(IInteractable interactable, Vector3 position)
    {
        SetDestination(position);
        
        // Aguarda chegar e interage
        StartCoroutine(WaitAndInteract(interactable, position));
    }

    private System.Collections.IEnumerator WaitAndInteract(IInteractable interactable, Vector3 position)
    {
        yield return new WaitUntil(() => 
            Vector3.Distance(transform.position, position) <= interactionRange || !IsMoving);
        
        if (Vector3.Distance(transform.position, position) <= interactionRange)
        {
            interactable.OnInteract(this);
        }
    }

    public void StopMovement()
    {
        agent.ResetPath();
        currentTarget = null;
        agent.stoppingDistance = stoppingDistance;
        
        // Reseta animação
        targetSpeed = 0f;
        currentSpeed = 0f;
        anim.SetFloat(moveSpeedParam, 0f);
        anim.SetBool(isMovingParam, false);
    }

    public void SetMovementEnabled(bool enabled)
    {
        isMovementEnabled = enabled;
        if (!enabled)
        {
            StopMovement();
        }
    }
    #endregion

    #region Atualização de Estado
    private void UpdateMovementState()
    {
        // Atualiza velocidade baseada em running
        float targetMoveSpeed = isRunning ? baseMoveSpeed * runSpeedMultiplier : baseMoveSpeed;
        
        // Aplica modificador de stats
        if (stats != null)
        {
            targetMoveSpeed *= (stats.MoveSpeed / 5f); // 5 é o base
        }
        
        agent.speed = targetMoveSpeed;
        
        // Calcula velocidade atual para blend tree
        if (agent.hasPath && agent.remainingDistance > agent.stoppingDistance)
        {
            targetSpeed = agent.velocity.magnitude / (baseMoveSpeed * runSpeedMultiplier);
        }
        else
        {
            targetSpeed = 0f;
        }
        
        // Smooth transition
        currentSpeed = Mathf.MoveTowards(currentSpeed, targetSpeed, 5f * Time.deltaTime);
    }

    private void UpdateAnimation()
    {
        bool wasMoving = anim.GetBool(isMovingParam);
        bool isMovingNow = IsMoving;
        
        anim.SetFloat(moveSpeedParam, currentSpeed);
        anim.SetBool(isMovingParam, isMovingNow);
        
        // Eventos de transição
        if (!wasMoving && isMovingNow)
        {
            OnMovementStarted?.Invoke();
        }
        else if (wasMoving && !isMovingNow)
        {
            OnMovementStopped?.Invoke();
            
            // Random idle variation
            anim.SetFloat(idleParam, Random.Range(0f, 1f));
        }
        
        // Rotação suave para direção do movimento
        if (isMovingNow && agent.velocity.sqrMagnitude > 0.01f)
        {
            Quaternion targetRotation = Quaternion.LookRotation(agent.velocity);
            transform.rotation = Quaternion.Slerp(transform.rotation, targetRotation, rotationSpeed * Time.deltaTime);
        }
    }
    #endregion

    #region Efeitos Visuais
    private void ShowClickEffect(Vector3 position)
    {
        // Instancia efeito de clique no chão (opcional)
        // ClickEffectManager.Instance?.Show(position);
    }
    #endregion

    #region Utilidades
    public void Teleport(Vector3 position)
    {
        agent.Warp(position);
        StopMovement();
    }

    public void LookAt(Vector3 target)
    {
        Vector3 direction = target - transform.position;
        direction.y = 0;
        if (direction.sqrMagnitude > 0.01f)
        {
            transform.rotation = Quaternion.LookRotation(direction);
        }
    }

    void OnDrawGizmosSelected()
    {
        if (agent != null && agent.hasPath)
        {
            Gizmos.color = Color.cyan;
            Vector3[] corners = agent.path.corners;
            for (int i = 0; i < corners.Length - 1; i++)
            {
                Gizmos.DrawLine(corners[i], corners[i + 1]);
            }
            
            Gizmos.color = Color.yellow;
            Gizmos.DrawWireSphere(agent.destination, 0.3f);
        }
        
        Gizmos.color = Color.green;
        Gizmos.DrawWireSphere(transform.position, interactionRange);
    }
    #endregion
}

/// <summary>
/// Interface para objetos interagíveis no mundo (NPCs, portais, baús, etc.)
/// </summary>
public interface IInteractable
{
    void OnInteract(PlayerMovement player);
    string GetInteractionName();
    float GetInteractionRange();
    bool CanInteract(PlayerMovement player);
}
'''

with open(f"{base_path}/Scripts/Player/PlayerMovement.cs", "w") as f:
    f.write(player_movement)

print("✅ PlayerMovement.cs criado!")
