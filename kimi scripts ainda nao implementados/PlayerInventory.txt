
# Script 6: PlayerInventory.cs - Inventário estilo TOP (grid 6x8)

player_inventory = '''using UnityEngine;
using Mirror;
using System.Collections.Generic;

/// <summary>
/// Sistema de Inventário fiel ao Tales of Pirates
/// Grid 6x8, empilhamento, itens equipáveis, gold, gems
/// </summary>
public class PlayerInventory : NetworkBehaviour
{
    #region Configuração
    [Header("Grid")]
    [SerializeField] private int gridWidth = 6;  // TOP usa 6 colunas
    [SerializeField] private int gridHeight = 8; // TOP usa 8 linhas
    
    [Header("Limites")]
    [SerializeField] private int maxGold = 999999999;
    [SerializeField] private int maxStackSize = 99; // TOP padrão
    [SerializeField] private int maxGems = 9999;
    #endregion

    #region Dados
    private SyncList<InventorySlot> inventorySlots = new SyncList<InventorySlot>();
    
    [SyncVar(hook = nameof(OnGoldChanged))]
    private int _gold;
    
    [SyncVar(hook = nameof(OnGemsChanged))]
    private int _gems;
    
    [SyncVar]
    private int _inventorySize; // Tamanho atual (pode expandir)
    #endregion

    #region Estado
    private int totalSlots;
    private CharacterStats stats;
    private PlayerEquipment equipment;
    #endregion

    #region Eventos
    public event System.Action<int> OnGoldChangedEvent;
    public event System.Action<int> OnGemsChangedEvent;
    public event System.Action<int, InventorySlot> OnItemAdded; // slot index, item
    public event System.Action<int, InventorySlot> OnItemRemoved;
    public event System.Action<int, InventorySlot> OnItemMoved; // from, to
    public event System.Action<int, int> OnItemStacked; // from, to
    public event System.Action OnInventoryFull;
    #endregion

    #region Propriedades
    public int Gold => _gold;
    public int Gems => _gems;
    public int GridWidth => gridWidth;
    public int GridHeight => gridHeight;
    public int TotalSlots => totalSlots;
    public IReadOnlyList<InventorySlot> Slots => inventorySlots;
    public bool IsFull => GetFirstEmptySlot() < 0;
    #endregion

    void Awake()
    {
        stats = GetComponent<CharacterStats>();
        equipment = GetComponent<PlayerEquipment>();
        totalSlots = gridWidth * gridHeight;
    }

    void Start()
    {
        if (isServer)
        {
            InitializeInventory();
        }
    }

    #region Inicialização
    [Server]
    private void InitializeInventory()
    {
        for (int i = 0; i < totalSlots; i++)
        {
            inventorySlots.Add(new InventorySlot());
        }
        
        _inventorySize = totalSlots;
        _gold = 0;
        _gems = 0;
    }
    #endregion

    #region Adicionar Itens
    [Server]
    public bool AddItem(ItemData item, int quantity = 1)
    {
        if (item == null || quantity <= 0) return false;
        
        int remaining = quantity;
        
        // Tenta empilhar primeiro
        if (item.isStackable)
        {
            for (int i = 0; i < totalSlots && remaining > 0; i++)
            {
                if (inventorySlots[i].itemId == item.itemId && inventorySlots[i].quantity < maxStackSize)
                {
                    int canAdd = Mathf.Min(remaining, maxStackSize - inventorySlots[i].quantity);
                    var slot = inventorySlots[i];
                    slot.quantity += canAdd;
                    inventorySlots[i] = slot;
                    remaining -= canAdd;
                }
            }
        }
        
        // Preenche slots vazios
        while (remaining > 0)
        {
            int emptySlot = GetFirstEmptySlot();
            if (emptySlot < 0)
            {
                // Inventário cheio
                OnInventoryFull?.Invoke();
                return false;
            }
            
            int addAmount = item.isStackable ? Mathf.Min(remaining, maxStackSize) : 1;
            
            var newSlot = new InventorySlot
            {
                itemId = item.itemId,
                quantity = addAmount,
                durability = item.maxDurability,
                isBound = item.isBound
            };
            
            inventorySlots[emptySlot] = newSlot;
            remaining -= addAmount;
            
            OnItemAdded?.Invoke(emptySlot, newSlot);
        }
        
        return true;
    }

    [Server]
    public bool AddItemToSlot(ItemData item, int slotIndex, int quantity = 1)
    {
        if (item == null || !IsValidSlot(slotIndex)) return false;
        if (!inventorySlots[slotIndex].IsEmpty) return false;
        
        var slot = new InventorySlot
        {
            itemId = item.itemId,
            quantity = quantity,
            durability = item.maxDurability,
            isBound = item.isBound
        };
        
        inventorySlots[slotIndex] = slot;
        OnItemAdded?.Invoke(slotIndex, slot);
        
        return true;
    }
    #endregion

    #region Remover Itens
    [Server]
    public bool RemoveItem(int slotIndex, int quantity = 1)
    {
        if (!IsValidSlot(slotIndex)) return false;
        if (inventorySlots[slotIndex].IsEmpty) return false;
        
        var slot = inventorySlots[slotIndex];
        
        if (slot.quantity < quantity) return false;
        
        slot.quantity -= quantity;
        
        if (slot.quantity <= 0)
        {
            inventorySlots[slotIndex] = new InventorySlot();
            OnItemRemoved?.Invoke(slotIndex, slot);
        }
        else
        {
            inventorySlots[slotIndex] = slot;
        }
        
        return true;
    }

    [Server]
    public bool RemoveItemById(string itemId, int quantity = 1)
    {
        int remaining = quantity;
        
        for (int i = 0; i < totalSlots && remaining > 0; i++)
        {
            if (inventorySlots[i].itemId == itemId)
            {
                int removeAmount = Mathf.Min(remaining, inventorySlots[i].quantity);
                RemoveItem(i, removeAmount);
                remaining -= removeAmount;
            }
        }
        
        return remaining <= 0;
    }
    #endregion

    #region Mover e Trocar
    [Server]
    public bool MoveItem(int fromSlot, int toSlot)
    {
        if (!IsValidSlot(fromSlot) || !IsValidSlot(toSlot)) return false;
        if (fromSlot == toSlot) return true;
        
        var from = inventorySlots[fromSlot];
        var to = inventorySlots[toSlot];
        
        // Slot destino vazio - apenas move
        if (to.IsEmpty)
        {
            inventorySlots[toSlot] = from;
            inventorySlots[fromSlot] = new InventorySlot();
            OnItemMoved?.Invoke(fromSlot, from);
            return true;
        }
        
        // Mesmo item e stackable - empilha
        if (from.itemId == to.itemId)
        {
            ItemData item = ItemDatabase.Instance?.GetItem(from.itemId);
            if (item != null && item.isStackable)
            {
                int canStack = Mathf.Min(from.quantity, maxStackSize - to.quantity);
                
                if (canStack > 0)
                {
                    to.quantity += canStack;
                    from.quantity -= canStack;
                    
                    inventorySlots[toSlot] = to;
                    
                    if (from.quantity <= 0)
                    {
                        inventorySlots[fromSlot] = new InventorySlot();
                    }
                    else
                    {
                        inventorySlots[fromSlot] = from;
                    }
                    
                    OnItemStacked?.Invoke(fromSlot, toSlot);
                    return true;
                }
            }
        }
        
        // Troca itens
        inventorySlots[fromSlot] = to;
        inventorySlots[toSlot] = from;
        OnItemMoved?.Invoke(fromSlot, from);
        
        return true;
    }

    [Server]
    public bool SplitStack(int slotIndex, int amount)
    {
        if (!IsValidSlot(slotIndex)) return false;
        
        var slot = inventorySlots[slotIndex];
        if (slot.quantity <= amount) return false;
        
        int emptySlot = GetFirstEmptySlot();
        if (emptySlot < 0) return false;
        
        slot.quantity -= amount;
        inventorySlots[slotIndex] = slot;
        
        var newSlot = new InventorySlot
        {
            itemId = slot.itemId,
            quantity = amount,
            durability = slot.durability,
            isBound = slot.isBound
        };
        
        inventorySlots[emptySlot] = newSlot;
        OnItemAdded?.Invoke(emptySlot, newSlot);
        
        return true;
    }
    #endregion

    #region Usar Itens
    [Server]
    public bool UseItem(int slotIndex)
    {
        if (!IsValidSlot(slotIndex)) return false;
        if (inventorySlots[slotIndex].IsEmpty) return false;
        
        var slot = inventorySlots[slotIndex];
        ItemData item = ItemDatabase.Instance?.GetItem(slot.itemId);
        
        if (item == null) return false;
        
        // Verifica se pode usar
        if (item.requiredLevel > stats.Level) return false;
        
        switch (item.itemType)
        {
            case ItemType.Consumable:
                return UseConsumable(slotIndex, item);
                
            case ItemType.Equipment:
                return EquipItem(slotIndex, item);
                
            case ItemType.SkillBook:
                return UseSkillBook(slotIndex, item);
                
            case ItemType.Quest:
                // Lógica de quest
                return false;
                
            default:
                return false;
        }
    }

    [Server]
    private bool UseConsumable(int slotIndex, ItemData item)
    {
        var slot = inventorySlots[slotIndex];
        
        // Aplica efeitos
        foreach (var effect in item.consumableEffects)
        {
            ApplyConsumableEffect(effect);
        }
        
        // Consome item
        slot.quantity--;
        if (slot.quantity <= 0)
        {
            inventorySlots[slotIndex] = new InventorySlot();
            OnItemRemoved?.Invoke(slotIndex, slot);
        }
        else
        {
            inventorySlots[slotIndex] = slot;
        }
        
        return true;
    }

    [Server]
    private bool EquipItem(int slotIndex, ItemData item)
    {
        EquipmentData equipData = item as EquipmentData;
        if (equipData == null) return false;
        
        if (equipment.EquipItem(equipData))
        {
            // Remove do inventário
            var slot = inventorySlots[slotIndex];
            slot.quantity--;
            
            if (slot.quantity <= 0)
            {
                inventorySlots[slotIndex] = new InventorySlot();
                OnItemRemoved?.Invoke(slotIndex, slot);
            }
            else
            {
                inventorySlots[slotIndex] = slot;
            }
            
            return true;
        }
        
        return false;
    }

    [Server]
    private bool UseSkillBook(int slotIndex, ItemData item)
    {
        SkillBookData skillBook = item as SkillBookData;
        if (skillBook == null) return false;
        
        PlayerSkills skills = GetComponent<PlayerSkills>();
        if (skills == null) return false;
        
        if (skills.LearnSkill(skillBook.skillData))
        {
            // Remove skill book
            var slot = inventorySlots[slotIndex];
            slot.quantity--;
            
            if (slot.quantity <= 0)
            {
                inventorySlots[slotIndex] = new InventorySlot();
            }
            else
            {
                inventorySlots[slotIndex] = slot;
            }
            
            return true;
        }
        
        return false;
    }

    [Server]
    private void ApplyConsumableEffect(ConsumableEffect effect)
    {
        switch (effect.effectType)
        {
            case ConsumableEffectType.Heal:
                stats.Heal(effect.value);
                break;
            case ConsumableEffectType.RestoreMana:
                stats.RestoreMana(effect.value);
                break;
            case ConsumableEffectType.RestoreStamina:
                stats.RestoreStamina(effect.value);
                break;
            case ConsumableEffectType.Buff:
                // Aplica buff temporário
                break;
            case ConsumableEffectType.Teleport:
                // Teleporta para cidade
                break;
            case ConsumableEffectType.Revive:
                // Ressurreição
                break;
        }
    }
    #endregion

    #region Gold e Gems
    [Server]
    public bool AddGold(int amount)
    {
        if (amount <= 0) return false;
        
        long newGold = (long)_gold + amount;
        if (newGold > maxGold)
        {
            _gold = maxGold;
            return false; // Não cabe tudo
        }
        
        _gold += amount;
        return true;
    }

    [Server]
    public bool RemoveGold(int amount)
    {
        if (amount <= 0 || _gold < amount) return false;
        _gold -= amount;
        return true;
    }

    [Server]
    public bool AddGems(int amount)
    {
        if (amount <= 0) return false;
        
        long newGems = (long)_gems + amount;
        if (newGems > maxGems)
        {
            _gems = maxGems;
            return false;
        }
        
        _gems += amount;
        return true;
    }

    [Server]
    public bool RemoveGems(int amount)
    {
        if (amount <= 0 || _gems < amount) return false;
        _gems -= amount;
        return true;
    }
    #endregion

    #region Utilidades
    private bool IsValidSlot(int index)
    {
        return index >= 0 && index < totalSlots;
    }

    private int GetFirstEmptySlot()
    {
        for (int i = 0; i < totalSlots; i++)
        {
            if (inventorySlots[i].IsEmpty) return i;
        }
        return -1;
    }

    public int GetItemCount(string itemId)
    {
        int count = 0;
        for (int i = 0; i < totalSlots; i++)
        {
            if (inventorySlots[i].itemId == itemId)
            {
                count += inventorySlots[i].quantity;
            }
        }
        return count;
    }

    public bool HasItem(string itemId, int quantity = 1)
    {
        return GetItemCount(itemId) >= quantity;
    }

    public InventorySlot GetSlot(int index)
    {
        if (!IsValidSlot(index)) return new InventorySlot();
        return inventorySlots[index];
    }
    #endregion

    #region Hooks
    private void OnGoldChanged(int oldVal, int newVal)
    {
        OnGoldChangedEvent?.Invoke(newVal);
    }

    private void OnGemsChanged(int oldVal, int newVal)
    {
        OnGemsChangedEvent?.Invoke(newVal);
    }
    #endregion
}

#region Structs e Enums
[System.Serializable]
public struct InventorySlot : Mirror.NetworkMessage
{
    public string itemId;
    public int quantity;
    public int durability;
    public bool isBound;
    public List<int> gems; // Para itens com sockets
    
    public bool IsEmpty => string.IsNullOrEmpty(itemId);
}

public enum ItemType
{
    Consumable,
    Equipment,
    Material,
    Quest,
    SkillBook,
    Gem,
    Currency,
    Pet,
    Mount,
    Costume,
    Misc
}

public enum ConsumableEffectType
{
    Heal,
    RestoreMana,
    RestoreStamina,
    Buff,
    Teleport,
    Revive,
    Experience,
    ResetStats
}

[System.Serializable]
public class ConsumableEffect
{
    public ConsumableEffectType effectType;
    public int value;
    public float duration; // 0 = instantâneo
    public StatType buffStat; // Para buffs
}
#endregion
'''

with open(f"{base_path}/Scripts/Inventory/PlayerInventory.cs", "w") as f:
    f.write(player_inventory)

print("✅ PlayerInventory.cs criado!")
