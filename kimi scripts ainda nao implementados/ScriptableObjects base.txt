# ItemData.cs
item_data = '''using UnityEngine;

/// <summary>
/// ScriptableObject base para todos os itens do Tales of Pirates
/// </summary>
[CreateAssetMenu(fileName = "NewItem", menuName = "TOP/Item")]
public class ItemData : ScriptableObject
{
    [Header("Informações Básicas")]
    public string itemId;
    public string itemName;
    [TextArea(3, 5)] public string description;
    public Sprite icon;
    public ItemType itemType;
    public ItemRarity rarity;
    
    [Header("Restrições")]
    public int requiredLevel;
    public CharacterClass requiredClass;
    public Gender requiredGender;
    public bool isBound;
    public bool isBoundOnEquip;
    public bool isTradable;
    public bool isDroppable;
    
    [Header("Propriedades")]
    public bool isStackable;
    public int maxStackSize = 99;
    public int maxDurability = 100;
    public int itemLevel; // Nível do item (para refinamento)
    public int sellPrice;
    public int buyPrice;
    
    [Header("Visual")]
    public GameObject modelPrefab;
    public Vector3 localPosition = Vector3.zero;
    public Vector3 localRotation = Vector3.zero;
    public Vector3 localScale = Vector3.one;
    
    [Header("Efeitos")]
    public AudioClip useSFX;
    public ParticleSystem useVFX;
    
    private void OnValidate()
    {
        if (string.IsNullOrEmpty(itemId))
        {
            itemId = System.Guid.NewGuid().ToString("N").Substring(0, 8);
        }
    }
}

public enum ItemRarity
{
    Common,      // Branco
    Uncommon,    // Verde
    Rare,        // Azul
    Epic,        // Roxo
    Legendary,   // Dourado
    Mythic       // Vermelho
}
'''

# SkillData.cs
skill_data = '''using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// ScriptableObject para skills do Tales of Pirates
/// </summary>
[CreateAssetMenu(fileName = "NewSkill", menuName = "TOP/Skill")]
public class SkillData : ScriptableObject
{
    [Header("Informações")]
    public string skillId;
    public string skillName;
    [TextArea(3, 5)] public string description;
    public Sprite icon;
    public SkillType skillType;
    public TargetType targetType;
    
    [Header("Requisitos")]
    public int requiredLevel;
    public CharacterClass requiredClass;
    public int skillPointCost = 1;
    public int upgradeCost = 1;
    public List<string> requiredSkills = new List<string>(); // Pré-requisitos
    public int maxLevel = 10;
    
    [Header("Custo")]
    public int manaCost;
    public int staminaCost;
    public float healthCost; // % de HP
    
    [Header("Casting")]
    public float castTime;
    public float cooldown;
    public float range;
    public float areaRadius; // Para skills de área
    public float coneAngle; // Para skills em cone
    
    [Header("Dano/Efeito")]
    public int baseDamage;
    public float damageMultiplier = 1f;
    public DamageType damageType;
    public List<StatusEffect> applyEffects = new List<StatusEffect>();
    
    [Header("Animação")]
    public int animationId;
    public bool spawnAtTarget;
    public bool spawnAtSelf;
    
    [Header("VFX/SFX")]
    public GameObject vfxPrefab;
    public ParticleSystem castVFX;
    public AudioClip sfxClip;
    public float cameraShake;
    public float cameraShakeDuration;
    
    [Header("Passiva")]
    public bool isPassive;
    public List<StatModifier> passiveModifiers = new List<StatModifier>();
    
    // Método de execução no servidor
    public virtual void ExecuteServer(PlayerSkills caster, CharacterStats target, Vector3 position, Quaternion rotation)
    {
        // Override em subclasses específicas
        if (target != null && baseDamage > 0)
        {
            int finalDamage = Mathf.FloorToInt(baseDamage * damageMultiplier * caster.GetComponent<CharacterStats>().MagicAttack / 10f);
            target.TakeDamage(finalDamage, caster.GetComponent<NetworkIdentity>().netId, damageType);
        }
        
        // Aplica efeitos
        foreach (var effect in applyEffects)
        {
            target?.GetComponent<CharacterStats>()?.ApplyStatusEffect(effect);
        }
    }
    
    private void OnValidate()
    {
        if (string.IsNullOrEmpty(skillId))
        {
            skillId = System.Guid.NewGuid().ToString("N").Substring(0, 8);
        }
    }
}
'''
