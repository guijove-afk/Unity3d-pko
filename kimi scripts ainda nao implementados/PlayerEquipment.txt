
# Script 4: PlayerEquipment.cs - Sistema de equipamentos completo

player_equipment = '''using UnityEngine;
using Mirror;
using System.Collections.Generic;

/// <summary>
/// Sistema de equipamentos fiel ao Tales of Pirates
/// Suporta: 10 slots, durabilidade, gems, refinamento, set bonuses
/// </summary>
public class PlayerEquipment : NetworkBehaviour
{
    #region Configuração
    [Header("Bones do Personagem")]
    [SerializeField] private Transform headBone;
    [SerializeField] private Transform bodyBone;
    [SerializeField] private Transform rightHandBone;
    [SerializeField] private Transform leftHandBone;
    [SerializeField] private Transform backBone;
    [SerializeField] private Transform feetBone;
    [SerializeField] private Transform handsBone;
    [SerializeField] private Transform neckBone;
    [SerializeField] private Transform ringLeftBone;
    [SerializeField] private Transform ringRightBone;
    
    [Header("Configurações")]
    [SerializeField] private bool showEquipmentModels = true;
    [SerializeField] private bool useAddressables = false;
    #endregion

    #region Equipamentos
    // TOP usa 10 slots principais
    private SyncList<EquipmentItem> equippedItems = new SyncList<EquipmentItem>();
    
    private Dictionary<EquipmentSlot, GameObject> currentModels = new Dictionary<EquipmentSlot, GameObject>();
    private CharacterStats stats;
    private List<StatModifier> equipmentModifiers = new List<StatModifier>();
    #endregion

    #region Eventos
    public event System.Action<EquipmentSlot, EquipmentData> OnItemEquipped;
    public event System.Action<EquipmentSlot> OnItemUnequipped;
    public event System.Action<EquipmentSlot> OnItemBroken; // Durabilidade zerou
    public event System.Action OnSetBonusChanged;
    #endregion

    #region Propriedades
    public IReadOnlyList<EquipmentItem> EquippedItems => equippedItems;
    
    public EquipmentData GetEquippedItem(EquipmentSlot slot)
    {
        int index = (int)slot;
        if (index < equippedItems.Count && !string.IsNullOrEmpty(equippedItems[index].itemId))
        {
            return ItemDatabase.Instance?.GetEquipment(equippedItems[index].itemId);
        }
        return null;
    }
    
    public bool HasItemEquipped(EquipmentSlot slot)
    {
        return GetEquippedItem(slot) != null;
    }
    #endregion

    void Awake()
    {
        stats = GetComponent<CharacterStats>();
        
        // Inicializa lista com 10 slots vazios
        for (int i = 0; i < 10; i++)
        {
            equippedItems.Add(new EquipmentItem());
        }
    }

    #region Equipar/Desequipar
    [Server]
    public bool EquipItem(EquipmentData item)
    {
        if (item == null) return false;
        
        // Verifica requisitos
        if (!CanEquip(item))
        {
            TargetShowError(connectionToClient, "Não atende os requisitos para equipar este item.");
            return false;
        }
        
        EquipmentSlot slot = item.equipmentSlot;
        int slotIndex = (int)slot;
        
        // Desequipa item atual
        if (HasItemEquipped(slot))
        {
            UnequipItem(slot);
        }
        
        // Equipa novo
        equippedItems[slotIndex] = new EquipmentItem
        {
            itemId = item.itemId,
            durability = item.maxDurability,
            refinement = 0,
            gems = new List<int>(),
            isBound = item.isBoundOnEquip
        };
        
        // Aplica modificadores de stats
        ApplyEquipmentStats(item);
        
        // Verifica set bonus
        CheckSetBonus();
        
        // Sincroniza visual
        RpcEquipVisual(slot, item.itemId);
        
        OnItemEquipped?.Invoke(slot, item);
        
        return true;
    }

    [Server]
    public void UnequipItem(EquipmentSlot slot)
    {
        int slotIndex = (int)slot;
        if (slotIndex >= equippedItems.Count) return;
        
        var equipped = equippedItems[slotIndex];
        if (string.IsNullOrEmpty(equipped.itemId)) return;
        
        EquipmentData item = ItemDatabase.Instance?.GetEquipment(equipped.itemId);
        if (item != null)
        {
            RemoveEquipmentStats(item);
        }
        
        equippedItems[slotIndex] = new EquipmentItem();
        
        CheckSetBonus();
        
        RpcUnequipVisual(slot);
        
        OnItemUnequipped?.Invoke(slot);
    }

    [Server]
    public bool CanEquip(EquipmentData item)
    {
        if (item == null) return false;
        
        // Level requirement
        if (stats != null && item.requiredLevel > stats.Level)
        {
            return false;
        }
        
        // Class requirement
        if (item.requiredClass != CharacterClass.None && item.requiredClass != stats.CharacterClass)
        {
            return false;
        }
        
        // Gender requirement (TOP tem restrição de gênero)
        if (item.requiredGender != Gender.Any && item.requiredGender != GetComponent<PlayerAppearance>()?.Gender)
        {
            return false;
        }
        
        // Durability check
        if (item.maxDurability <= 0)
        {
            return false;
        }
        
        return true;
    }
    #endregion

    #region Stats de Equipamento
    [Server]
    private void ApplyEquipmentStats(EquipmentData item)
    {
        // Stats base do item
        foreach (var mod in item.statModifiers)
        {
            equipmentModifiers.Add(mod);
            stats.AddModifier(mod);
        }
        
        // Stats de refinamento
        if (item.refinementStats != null)
        {
            int refinementLevel = GetEquippedItem(item.equipmentSlot)?.refinement ?? 0;
            foreach (var mod in item.refinementStats)
            {
                if (refinementLevel >= mod.requiredRefinement)
                {
                    equipmentModifiers.Add(mod.modifier);
                    stats.AddModifier(mod.modifier);
                }
            }
        }
        
        // Stats de gems
        var equipped = equippedItems[(int)item.equipmentSlot];
        foreach (int gemId in equipped.gems)
        {
            GemData gem = ItemDatabase.Instance?.GetGem(gemId);
            if (gem != null)
            {
                equipmentModifiers.Add(gem.statModifier);
                stats.AddModifier(gem.statModifier);
            }
        }
    }

    [Server]
    private void RemoveEquipmentStats(EquipmentData item)
    {
        // Remove todos os modificadores deste item
        List<StatModifier> toRemove = new List<StatModifier>();
        
        foreach (var mod in equipmentModifiers)
        {
            if (mod.source == item.itemId)
            {
                toRemove.Add(mod);
                stats.RemoveModifier(mod);
            }
        }
        
        foreach (var mod in toRemove)
        {
            equipmentModifiers.Remove(mod);
        }
    }
    #endregion

    #region Durabilidade
    [Server]
    public void DamageDurability(EquipmentSlot slot, int amount)
    {
        int index = (int)slot;
        if (index >= equippedItems.Count) return;
        
        var item = equippedItems[index];
        if (string.IsNullOrEmpty(item.itemId)) return;
        
        item.durability -= amount;
        
        if (item.durability <= 0)
        {
            item.durability = 0;
            OnItemBroken?.Invoke(slot);
            
            // Item quebrado - remove stats mas mantém equipado (visível mas sem bonus)
            EquipmentData data = ItemDatabase.Instance?.GetEquipment(item.itemId);
            if (data != null)
            {
                RemoveEquipmentStats(data);
            }
        }
        
        equippedItems[index] = item;
    }

    [Server]
    public void RepairItem(EquipmentSlot slot)
    {
        int index = (int)slot;
        if (index >= equippedItems.Count) return;
        
        var item = equippedItems[index];
        if (string.IsNullOrEmpty(item.itemId)) return;
        
        EquipmentData data = ItemDatabase.Instance?.GetEquipment(item.itemId);
        if (data != null)
        {
            item.durability = data.maxDurability;
            equippedItems[index] = item;
            
            // Reaplica stats
            ApplyEquipmentStats(data);
        }
    }
    #endregion

    #region Refinamento
    [Server]
    public bool RefineItem(EquipmentSlot slot)
    {
        int index = (int)slot;
        if (index >= equippedItems.Count) return false;
        
        var item = equippedItems[index];
        EquipmentData data = ItemDatabase.Instance?.GetEquipment(item.itemId);
        
        if (data == null || item.refinement >= data.maxRefinement)
        {
            return false;
        }
        
        // Chance de sucesso baseada no nível de refinamento
        float successChance = GetRefineSuccessChance(item.refinement);
        
        if (Random.Range(0f, 1f) <= successChance)
        {
            item.refinement++;
            equippedItems[index] = item;
            
            // Aplica novos stats de refinamento
            RemoveEquipmentStats(data);
            ApplyEquipmentStats(data);
            
            return true;
        }
        else
        {
            // Falha - pode quebrar item dependendo do nível
            if (item.refinement >= 4 && Random.Range(0f, 1f) < 0.3f)
            {
                // Item quebra!
                UnequipItem(slot);
            }
            
            return false;
        }
    }

    private float GetRefineSuccessChance(int currentLevel)
    {
        // Chances do TOP
        return currentLevel switch
        {
            0 => 1.0f,   // 100%
            1 => 0.9f,   // 90%
            2 => 0.8f,   // 80%
            3 => 0.7f,   // 70%
            4 => 0.6f,   // 60%
            5 => 0.5f,   // 50%
            6 => 0.4f,   // 40%
            7 => 0.3f,   // 30%
            8 => 0.2f,   // 20%
            9 => 0.1f,   // 10%
            _ => 0.05f   // 5%
        };
    }
    #endregion

    #region Gems
    [Server]
    public bool AddGem(EquipmentSlot slot, int gemId)
    {
        int index = (int)slot;
        if (index >= equippedItems.Count) return false;
        
        var item = equippedItems[index];
        EquipmentData data = ItemDatabase.Instance?.GetEquipment(item.itemId);
        
        if (data == null) return false;
        if (item.gems.Count >= data.maxGems) return false;
        
        GemData gem = ItemDatabase.Instance?.GetGem(gemId);
        if (gem == null) return false;
        
        item.gems.Add(gemId);
        equippedItems[index] = item;
        
        // Aplica stats da gem
        equipmentModifiers.Add(gem.statModifier);
        stats.AddModifier(gem.statModifier);
        
        return true;
    }

    [Server]
    public bool RemoveGem(EquipmentSlot slot, int gemIndex)
    {
        int index = (int)slot;
        if (index >= equippedItems.Count) return false;
        
        var item = equippedItems[index];
        if (gemIndex >= item.gems.Count) return false;
        
        int gemId = item.gems[gemIndex];
        GemData gem = ItemDatabase.Instance?.GetGem(gemId);
        
        item.gems.RemoveAt(gemIndex);
        equippedItems[index] = item;
        
        // Remove stats da gem
        if (gem != null)
        {
            equipmentModifiers.Remove(gem.statModifier);
            stats.RemoveModifier(gem.statModifier);
        }
        
        return true;
    }
    #endregion

    #region Set Bonus
    [Server]
    private void CheckSetBonus()
    {
        // Verifica se possui itens de um set completo
        Dictionary<string, int> setCounts = new Dictionary<string, int>();
        
        foreach (var equipped in equippedItems)
        {
            if (string.IsNullOrEmpty(equipped.itemId)) continue;
            
            EquipmentData data = ItemDatabase.Instance?.GetEquipment(equipped.itemId);
            if (data != null && !string.IsNullOrEmpty(data.setName))
            {
                if (!setCounts.ContainsKey(data.setName))
                    setCounts[data.setName] = 0;
                setCounts[data.setName]++;
            }
        }
        
        // Aplica bônus de set
        foreach (var kvp in setCounts)
        {
            SetData setData = ItemDatabase.Instance?.GetSetData(kvp.Key);
            if (setData != null)
            {
                foreach (var bonus in setData.bonuses)
                {
                    if (kvp.Value >= bonus.requiredPieces)
                    {
                        // Aplica bônus
                        foreach (var mod in bonus.modifiers)
                        {
                            stats.AddModifier(mod);
                        }
                    }
                }
            }
        }
        
        OnSetBonusChanged?.Invoke();
    }
    #endregion

    #region Visual (Client)
    [ClientRpc]
    private void RpcEquipVisual(EquipmentSlot slot, string itemId)
    {
        if (!showEquipmentModels) return;
        
        EquipmentData data = ItemDatabase.Instance?.GetEquipment(itemId);
        if (data == null || data.modelPrefab == null) return;
        
        Transform bone = GetBoneForSlot(slot);
        if (bone == null) return;
        
        // Remove modelo anterior
        if (currentModels.ContainsKey(slot))
        {
            Destroy(currentModels[slot]);
        }
        
        // Instancia novo modelo
        GameObject model = Instantiate(data.modelPrefab, bone);
        model.transform.localPosition = data.localPosition;
        model.transform.localRotation = Quaternion.Euler(data.localRotation);
        model.transform.localScale = data.localScale;
        
        currentModels[slot] = model;
    }

    [ClientRpc]
    private void RpcUnequipVisual(EquipmentSlot slot)
    {
        if (currentModels.ContainsKey(slot))
        {
            Destroy(currentModels[slot]);
            currentModels.Remove(slot);
        }
    }

    private Transform GetBoneForSlot(EquipmentSlot slot)
    {
        return slot switch
        {
            EquipmentSlot.Helmet => headBone,
            EquipmentSlot.Armor => bodyBone,
            EquipmentSlot.Weapon => rightHandBone,
            EquipmentSlot.Shield => leftHandBone,
            EquipmentSlot.Gloves => handsBone,
            EquipmentSlot.Boots => feetBone,
            EquipmentSlot.Cape => backBone,
            EquipmentSlot.Necklace => neckBone,
            EquipmentSlot.RingLeft => ringLeftBone,
            EquipmentSlot.RingRight => ringRightBone,
            _ => null
        };
    }
    #endregion

    #region Utilidades
    [TargetRpc]
    private void TargetShowError(NetworkConnection target, string message)
    {
        UIManager.Instance?.ShowErrorMessage(message);
    }

    public int GetTotalEquipmentScore()
    {
        int score = 0;
        foreach (var equipped in equippedItems)
        {
            if (string.IsNullOrEmpty(equipped.itemId)) continue;
            
            EquipmentData data = ItemDatabase.Instance?.GetEquipment(equipped.itemId);
            if (data != null)
            {
                score += data.itemLevel * 10;
                score += equipped.refinement * 5;
            }
        }
        return score;
    }
    #endregion
}

#region Structs e Enums
[System.Serializable]
public struct EquipmentItem : Mirror.NetworkMessage
{
    public string itemId;
    public int durability;
    public int refinement;
    public List<int> gems;
    public bool isBound;
}

public enum EquipmentSlot
{
    Helmet = 0,
    Armor = 1,
    Weapon = 2,
    Shield = 3,
    Gloves = 4,
    Boots = 5,
    Cape = 6,
    Necklace = 7,
    RingLeft = 8,
    RingRight = 9
}

public enum Gender
{
    Any,
    Male,
    Female
}

[System.Serializable]
public class RefinementStat
{
    public int requiredRefinement;
    public StatModifier modifier;
}

[System.Serializable]
public class SetBonus
{
    public int requiredPieces; // 2, 3, 5, etc.
    public List<StatModifier> modifiers;
    public string description;
}
#endregion
'''

with open(f"{base_path}/Scripts/Equipment/PlayerEquipment.cs", "w") as f:
    f.write(player_equipment)

print("✅ PlayerEquipment.cs criado!")
