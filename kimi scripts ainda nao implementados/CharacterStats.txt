
# Script 1: CharacterStats.cs - Sistema completo de stats fiel ao TOP

character_stats = '''using UnityEngine;
using Mirror;
using System;
using System.Collections.Generic;

/// <summary>
/// Sistema de Stats completo baseado em Tales of Pirates
/// Fórmulas originais adaptadas para Unity
/// </summary>
[RequireComponent(typeof(CharacterController))]
public class CharacterStats : NetworkBehaviour, IDamageable
{
    #region SyncVars - Stats Principais
    [SyncVar(hook = nameof(OnHealthChanged))] 
    private int _currentHealth;
    
    [SyncVar(hook = nameof(OnMaxHealthChanged))] 
    private int _maxHealth;
    
    [SyncVar(hook = nameof(OnManaChanged))] 
    private int _currentMana;
    
    [SyncVar(hook = nameof(OnMaxManaChanged))] 
    private int _maxMana;
    
    [SyncVar(hook = nameof(OnStaminaChanged))]
    private int _currentStamina;
    
    [SyncVar]
    private int _maxStamina;
    #endregion

    #region SyncVars - Atributos Base (TOP: STR, AGI, CON, SPR, ACC)
    [SyncVar] private int _strength;      // STR - Ataque físico, HP
    [SyncVar] private int _agility;       // AGI - Velocidade, evasão, atk speed
    [SyncVar] private int _constitution;  // CON - HP, defesa, stamina
    [SyncVar] private int _spirit;        // SPR - MP, ataque mágico, def mágica
    [SyncVar] private int _accuracy;      // ACC - Precisão, chance de hit
    [SyncVar] private int _luck;         // LUK - Crítico, drop rate
    #endregion

    #region SyncVars - Stats de Combate Derivados
    [SyncVar] private int _attack;
    [SyncVar] private int _defense;
    [SyncVar] private int _magicAttack;
    [SyncVar] private int _magicDefense;
    [SyncVar] private float _attackSpeed;
    [SyncVar] private float _moveSpeed;
    [SyncVar] private int _evasion;
    [SyncVar] private int _hitRate;
    [SyncVar] private float _criticalRate;
    [SyncVar] private float _criticalDamage;
    #endregion

    #region SyncVars - Progressão
    [SyncVar(hook = nameof(OnLevelChanged))] 
    private int _level = 1;
    
    [SyncVar] private int _experience;
    [SyncVar] private int _statPoints;    // Pontos para distribuir
    [SyncVar] private int _skillPoints;   // Pontos de skill
    [SyncVar] private int _reputation;    // Fama
    [SyncVar] private int _pkValue;       // Valor PK (player killer)
    #endregion

    #region SyncVars - Estado
    [SyncVar] private bool _isDead;
    [SyncVar] private bool _isStunned;
    [SyncVar] private bool _isSilenced;
    [SyncVar] private CharacterClass _characterClass;
    [SyncVar] private CharacterType _characterType;
    #endregion

    #region Propriedades Públicas
    public int CurrentHealth => _currentHealth;
    public int MaxHealth => _maxHealth;
    public int CurrentMana => _currentMana;
    public int MaxMana => _maxMana;
    public int CurrentStamina => _currentStamina;
    public int MaxStamina => _maxStamina;
    
    public int Strength => _strength;
    public int Agility => _agility;
    public int Constitution => _constitution;
    public int Spirit => _spirit;
    public int Accuracy => _accuracy;
    public int Luck => _luck;
    
    public int Attack => _attack;
    public int Defense => _defense;
    public int MagicAttack => _magicAttack;
    public int MagicDefense => _magicDefense;
    public float AttackSpeed => _attackSpeed;
    public float MoveSpeed => _moveSpeed;
    public int Evasion => _evasion;
    public int HitRate => _hitRate;
    public float CriticalRate => _criticalRate;
    public float CriticalDamage => _criticalDamage;
    
    public int Level => _level;
    public int Experience => _experience;
    public int StatPoints => _statPoints;
    public int SkillPoints => _skillPoints;
    public int Reputation => _reputation;
    public int PKValue => _pkValue;
    
    public bool IsDead => _isDead;
    public bool IsStunned => _isStunned;
    public bool IsSilenced => _isSilenced;
    public CharacterClass CharacterClass => _characterClass;
    public CharacterType CharacterType => _characterType;
    
    public float HealthPercentage => _maxHealth > 0 ? (float)_currentHealth / _maxHealth : 0;
    public float ManaPercentage => _maxMana > 0 ? (float)_currentMana / _maxMana : 0;
    #endregion

    #region Eventos
    public event Action<int, int> OnHealthChangedEvent;      // current, max
    public event Action<int, int> OnManaChangedEvent;
    public event Action<int, int> OnStaminaChangedEvent;
    public event Action<StatType, int> OnStatChanged;        // tipo, novo valor
    public event Action<int> OnLevelUp;                      // novo level
    public event Action<int> OnExperienceGained;             // XP ganho
    public event Action OnDeathEvent;
    public event Action OnReviveEvent;
    public event Action OnStatsRecalculated;
    #endregion

    #region Modificadores Temporários
    private List<StatModifier> _activeModifiers = new List<StatModifier>();
    private List<StatusEffect> _activeEffects = new List<StatusEffect>();
    #endregion

    #region Configuração
    [Header("Configuração de Classe")]
    [SerializeField] private ClassData classData;
    
    [Header("Configuração de Tipo")]
    [SerializeField] private CharacterType startingType = CharacterType.Player;
    #endregion

    void Awake()
    {
        _characterType = startingType;
    }

    void Start()
    {
        if (isServer)
        {
            InitializeStats();
        }
    }

    void Update()
    {
        if (isServer)
        {
            UpdateStatusEffects();
            UpdateRegeneration();
        }
    }

    #region Inicialização
    [Server]
    public void InitializeStats()
    {
        if (classData != null)
        {
            _characterClass = classData.characterClass;
            
            // Stats base da classe
            _strength = classData.baseStrength;
            _agility = classData.baseAgility;
            _constitution = classData.baseConstitution;
            _spirit = classData.baseSpirit;
            _accuracy = classData.baseAccuracy;
            _luck = classData.baseLuck;
        }
        else
        {
            // Stats padrão se não houver classe definida
            _strength = 10;
            _agility = 10;
            _constitution = 10;
            _spirit = 10;
            _accuracy = 10;
            _luck = 5;
        }

        RecalculateDerivedStats();
        
        _currentHealth = _maxHealth;
        _currentMana = _maxMana;
        _currentStamina = _maxStamina;
        _isDead = false;
    }
    #endregion

    #region Cálculo de Stats Derivados (Fórmulas TOP)
    [Server]
    public void RecalculateDerivedStats()
    {
        // HP: 100 + CON*10 + Level*5
        _maxHealth = 100 + (_constitution * 10) + (_level * 5);
        
        // MP: 50 + SPR*8 + Level*3
        _maxMana = 50 + (_spirit * 8) + (_level * 3);
        
        // Stamina: 100 + CON*5
        _maxStamina = 100 + (_constitution * 5);
        
        // Ataque Físico: STR*2 + AGI + Level
        _attack = (_strength * 2) + _agility + _level;
        
        // Defesa: CON + AGI/2 + Level/2
        _defense = _constitution + (_agility / 2) + (_level / 2);
        
        // Ataque Mágico: SPR*2 + Level
        _magicAttack = (_spirit * 2) + _level;
        
        // Defesa Mágica: SPR + CON/2
        _magicDefense = _spirit + (_constitution / 2);
        
        // Velocidade de Ataque: 1.0 + AGI*0.01
        _attackSpeed = 1.0f + (_agility * 0.01f);
        
        // Velocidade de Movimento: 5.0 + AGI*0.02
        _moveSpeed = 5.0f + (_agility * 0.02f);
        
        // Evasão: AGI + LUK/2
        _evasion = _agility + (_luck / 2);
        
        // Hit Rate: ACC + Level + AGI/2
        _hitRate = _accuracy + _level + (_agility / 2);
        
        // Taxa Crítica: LUK*0.1% (base)
        _criticalRate = _luck * 0.1f;
        
        // Dano Crítico: 150% + LUK*0.2%
        _criticalDamage = 1.5f + (_luck * 0.002f);

        // Aplicar modificadores de equipamentos/buffs
        ApplyModifiers();

        // Garantir que HP/MP não excedam novos máximos
        _currentHealth = Mathf.Min(_currentHealth, _maxHealth);
        _currentMana = Mathf.Min(_currentMana, _maxMana);
        _currentStamina = Mathf.Min(_currentStamina, _maxStamina);

        OnStatsRecalculated?.Invoke();
    }

    [Server]
    private void ApplyModifiers()
    {
        foreach (var modifier in _activeModifiers)
        {
            switch (modifier.statType)
            {
                case StatType.Strength: _strength += modifier.value; break;
                case StatType.Agility: _agility += modifier.value; break;
                case StatType.Constitution: _constitution += modifier.value; break;
                case StatType.Spirit: _spirit += modifier.value; break;
                case StatType.Accuracy: _accuracy += modifier.value; break;
                case StatType.Luck: _luck += modifier.value; break;
                case StatType.Attack: _attack += modifier.value; break;
                case StatType.Defense: _defense += modifier.value; break;
                case StatType.MaxHealth: _maxHealth += modifier.value; break;
                case StatType.MaxMana: _maxMana += modifier.value; break;
                case StatType.AttackSpeed: _attackSpeed += modifier.value * 0.01f; break;
                case StatType.MoveSpeed: _moveSpeed += modifier.value * 0.01f; break;
            }
        }
    }
    #endregion

    #region Dano e Cura
    [Server]
    public void TakeDamage(int damage, uint attackerId, DamageType damageType = DamageType.Physical)
    {
        if (_isDead || _isStunned) return;

        int finalDamage = CalculateFinalDamage(damage, damageType);
        
        _currentHealth = Mathf.Max(0, _currentHealth - finalDamage);
        
        // Notifica cliente do alvo
        TargetOnDamageTaken(connectionToClient, finalDamage, attackerId, damageType);
        
        // Notifica todos os clientes próximos (para VFX)
        RpcOnDamageTaken(finalDamage, transform.position);

        if (_currentHealth <= 0)
        {
            Die(attackerId);
        }
    }

    [Server]
    private int CalculateFinalDamage(int baseDamage, DamageType damageType)
    {
        int finalDamage = baseDamage;
        
        switch (damageType)
        {
            case DamageType.Physical:
                finalDamage = Mathf.Max(1, baseDamage - _defense);
                break;
            case DamageType.Magical:
                finalDamage = Mathf.Max(1, baseDamage - _magicDefense);
                break;
            case DamageType.True:
                // Dano verdadeiro ignora defesa
                break;
        }

        // Aplicar redução por buffs/debuffs
        foreach (var effect in _activeEffects)
        {
            if (effect.effectType == StatusEffectType.DamageReduction)
            {
                finalDamage = Mathf.FloorToInt(finalDamage * (1 - effect.magnitude));
            }
        }

        // Randomização de ±10%
        float randomFactor = UnityEngine.Random.Range(0.9f, 1.1f);
        finalDamage = Mathf.FloorToInt(finalDamage * randomFactor);

        return Mathf.Max(1, finalDamage);
    }

    [Server]
    public void Heal(int amount)
    {
        if (_isDead) return;
        _currentHealth = Mathf.Min(_maxHealth, _currentHealth + amount);
        RpcOnHealed(amount);
    }

    [Server]
    public void RestoreMana(int amount)
    {
        _currentMana = Mathf.Min(_maxMana, _currentMana + amount);
    }

    [Server]
    public void RestoreStamina(int amount)
    {
        _currentStamina = Mathf.Min(_maxStamina, _currentStamina + amount);
    }
    #endregion

    #region Morte e Ressurreição
    [Server]
    private void Die(uint killerId)
    {
        _isDead = true;
        _currentHealth = 0;
        
        // Penalidade de EXP (5% no TOP)
        int expLoss = Mathf.FloorToInt(_experience * 0.05f);
        _experience = Mathf.Max(0, _experience - expLoss);
        
        // PK System
        if (killerId != netId && killerId != 0)
        {
            // Lógica de PK
        }

        OnDeathEvent?.Invoke();
        RpcOnDeath();
    }

    [Server]
    public void Revive(bool atSpawn = true)
    {
        _isDead = false;
        _currentHealth = Mathf.FloorToInt(_maxHealth * 0.3f); // 30% HP ao reviver
        _currentMana = Mathf.FloorToInt(_maxMana * 0.3f);
        
        OnReviveEvent?.Invoke();
        RpcOnRevive();
    }
    #endregion

    #region Level Up
    [Server]
    public void AddExperience(int amount)
    {
        if (_isDead) return;
        
        _experience += amount;
        OnExperienceGained?.Invoke(amount);
        
        while (_experience >= GetExpForNextLevel())
        {
            LevelUp();
        }
    }

    [Server]
    private void LevelUp()
    {
        int expNeeded = GetExpForNextLevel();
        _experience -= expNeeded;
        _level++;
        
        // TOP: 5 pontos de status por level
        _statPoints += 5;
        
        // TOP: 1 ponto de skill por level
        _skillPoints += 1;
        
        // Recupera HP/MP completamente ao upar
        RecalculateDerivedStats();
        _currentHealth = _maxHealth;
        _currentMana = _maxMana;
        _currentStamina = _maxStamina;
        
        OnLevelUp?.Invoke(_level);
        RpcOnLevelUp(_level);
    }

    public int GetExpForNextLevel()
    {
        // Fórmula TOP: 100 * level^1.5
        return Mathf.FloorToInt(100 * Mathf.Pow(_level, 1.5f));
    }

    public int GetTotalExpForLevel(int targetLevel)
    {
        int total = 0;
        for (int i = 1; i < targetLevel; i++)
        {
            total += Mathf.FloorToInt(100 * Mathf.Pow(i, 1.5f));
        }
        return total;
    }
    #endregion

    #region Distribuição de Pontos
    [Server]
    public bool AddStatPoint(StatType stat)
    {
        if (_statPoints <= 0) return false;
        
        switch (stat)
        {
            case StatType.Strength: _strength++; break;
            case StatType.Agility: _agility++; break;
            case StatType.Constitution: _constitution++; break;
            case StatType.Spirit: _spirit++; break;
            case StatType.Accuracy: _accuracy++; break;
            case StatType.Luck: _luck++; break;
            default: return false;
        }
        
        _statPoints--;
        RecalculateDerivedStats();
        OnStatChanged?.Invoke(stat, GetStatValue(stat));
        
        return true;
    }

    public int GetStatValue(StatType stat)
    {
        return stat switch
        {
            StatType.Strength => _strength,
            StatType.Agility => _agility,
            StatType.Constitution => _constitution,
            StatType.Spirit => _spirit,
            StatType.Accuracy => _accuracy,
            StatType.Luck => _luck,
            StatType.Attack => _attack,
            StatType.Defense => _defense,
            StatType.MaxHealth => _maxHealth,
            StatType.MaxMana => _maxMana,
            _ => 0
        };
    }
    #endregion

    #region Modificadores e Efeitos
    [Server]
    public void AddModifier(StatModifier modifier)
    {
        _activeModifiers.Add(modifier);
        RecalculateDerivedStats();
    }

    [Server]
    public void RemoveModifier(StatModifier modifier)
    {
        _activeModifiers.Remove(modifier);
        RecalculateDerivedStats();
    }

    [Server]
    public void ApplyStatusEffect(StatusEffect effect)
    {
        _activeEffects.Add(effect);
        
        switch (effect.effectType)
        {
            case StatusEffectType.Stun: _isStunned = true; break;
            case StatusEffectType.Silence: _isSilenced = true; break;
        }
        
        RpcOnStatusEffectApplied(effect.effectType, effect.duration);
    }

    [Server]
    public void RemoveStatusEffect(StatusEffect effect)
    {
        _activeEffects.Remove(effect);
        
        // Verifica se ainda há efeitos do mesmo tipo
        bool hasStun = _activeEffects.Exists(e => e.effectType == StatusEffectType.Stun);
        bool hasSilence = _activeEffects.Exists(e => e.effectType == StatusEffectType.Silence);
        
        _isStunned = hasStun;
        _isSilenced = hasSilence;
    }

    [Server]
    private void UpdateStatusEffects()
    {
        for (int i = _activeEffects.Count - 1; i >= 0; i--)
        {
            _activeEffects[i].remainingDuration -= Time.deltaTime;
            
            if (_activeEffects[i].remainingDuration <= 0)
            {
                RemoveStatusEffect(_activeEffects[i]);
            }
        }
    }

    [Server]
    private void UpdateRegeneration()
    {
        // Regeneração natural (TOP: 1% HP/MP a cada 5 segundos)
        // Implementar com timer
    }
    #endregion

    #region RPCs
    [TargetRpc]
    private void TargetOnDamageTaken(NetworkConnection target, int damage, uint attackerId, DamageType damageType)
    {
        // Feedback local de dano recebido
        DamagePopupManager.Instance?.ShowDamage(transform.position + Vector3.up * 2f, damage, true);
        
        // Efeito de hit
        if (damageType == DamageType.Physical)
        {
            // Som de hit físico
        }
        else if (damageType == DamageType.Magical)
        {
            // Som de hit mágico
        }
    }

    [ClientRpc]
    private void RpcOnDamageTaken(int damage, Vector3 position)
    {
        // VFX de dano para observadores
        if (!isLocalPlayer)
        {
            DamagePopupManager.Instance?.ShowDamage(position + Vector3.up * 2f, damage, false);
        }
    }

    [ClientRpc]
    private void RpcOnHealed(int amount)
    {
        DamagePopupManager.Instance?.ShowHeal(transform.position + Vector3.up * 2f, amount);
    }

    [ClientRpc]
    private void RpcOnDeath()
    {
        GetComponent<Animator>()?.SetTrigger("Die");
        GetComponent<PlayerMovement>()?.SetMovementEnabled(false);
        GetComponent<PlayerCombat>()?.enabled = false;
        
        // Efeito de morte
        // Som de morte
    }

    [ClientRpc]
    private void RpcOnRevive()
    {
        GetComponent<Animator>()?.SetTrigger("Revive");
        GetComponent<PlayerMovement>()?.SetMovementEnabled(true);
        GetComponent<PlayerCombat>()?.enabled = true;
        
        // Efeito de revive
    }

    [ClientRpc]
    private void RpcOnLevelUp(int newLevel)
    {
        // Efeito visual de level up
        LevelUpEffectManager.Instance?.PlayEffect(transform.position);
        
        // Som de level up
        AudioManager.Instance?.PlaySFX("LevelUp");
        
        // Notificação na UI
        UIManager.Instance?.ShowLevelUpNotification(newLevel);
    }

    [ClientRpc]
    private void RpcOnStatusEffectApplied(StatusEffectType effectType, float duration)
    {
        // Ícone de debuff na UI
        UIManager.Instance?.ShowStatusEffect(effectType, duration);
    }
    #endregion

    #region Hooks SyncVar
    private void OnHealthChanged(int oldVal, int newVal)
    {
        OnHealthChangedEvent?.Invoke(newVal, _maxHealth);
    }

    private void OnMaxHealthChanged(int oldVal, int newVal)
    {
        OnHealthChangedEvent?.Invoke(_currentHealth, newVal);
    }

    private void OnManaChanged(int oldVal, int newVal)
    {
        OnManaChangedEvent?.Invoke(newVal, _maxMana);
    }

    private void OnMaxManaChanged(int oldVal, int newVal)
    {
        OnManaChangedEvent?.Invoke(_currentMana, newVal);
    }

    private void OnStaminaChanged(int oldVal, int newVal)
    {
        OnStaminaChangedEvent?.Invoke(newVal, _maxStamina);
    }

    private void OnLevelChanged(int oldVal, int newVal)
    {
        // Level up já é notificado por evento separado
    }
    #endregion

    #region Utilidades
    [Server]
    public void SetCharacterClass(CharacterClass newClass)
    {
        _characterClass = newClass;
        classData = ClassDatabase.Instance?.GetClassData(newClass);
        InitializeStats();
    }

    public void ResetStats()
    {
        if (!isServer) return;
        
        _statPoints += (_strength + _agility + _constitution + _spirit + _accuracy + _luck) - 50;
        
        _strength = 10;
        _agility = 10;
        _constitution = 10;
        _spirit = 10;
        _accuracy = 10;
        _luck = 5;
        
        RecalculateDerivedStats();
    }
    #endregion
}

#region Enums e Structs
public enum CharacterClass
{
    None,
    // Primeira Job
    Swordsman = 1,

    Explorer = 3,

    // Segunda Job
    Champion = 11,
    Crusader = 12,

    Cleric = 15
}

public enum CharacterType
{
    Player,
    NPC,
    Monster,
    Boss,
    Pet,
    Mount
}

public enum StatType
{
    Strength,
    Agility,
    Constitution,
    Spirit,
    Accuracy,
    Luck,
    Attack,
    Defense,
    MagicAttack,
    MagicDefense,
    MaxHealth,
    MaxMana,
    AttackSpeed,
    MoveSpeed,
    Evasion,
    HitRate,
    CriticalRate
}

public enum DamageType
{
    Physical,
    Magical,
    True,
    Poison,
    Fire,
    Ice,
    Lightning,
    Holy,
    Dark
}

[System.Serializable]
public struct StatModifier
{
    public StatType statType;
    public int value;
    public float duration; // 0 = permanente
    public string source; // equipamento, buff, etc.
}

[System.Serializable]
public class StatusEffect
{
    public StatusEffectType effectType;
    public float magnitude;
    public float duration;
    public float remainingDuration;
    public uint casterId;
    public bool isDebuff;
}

public enum StatusEffectType
{
    Stun,
    Silence,
    Poison,
    Burn,
    Freeze,
    Slow,
    Haste,
    DamageReduction,
    DamageBoost,
    HealOverTime,
    ManaRegen,
    Invisible,
    Invincible
}

public interface IDamageable
{
    void TakeDamage(int damage, uint attackerId, DamageType damageType);
    bool IsDead { get; }
}
#endregion
