
# Script 5: PlayerSkills.cs - Sistema de skills completo

player_skills = '''using UnityEngine;
using Mirror;
using System.Collections.Generic;

/// <summary>
/// Sistema de Skills completo baseado em Tales of Pirates
/// Skill bar F1-F10, cast time, cooldown, combos, área e target
/// </summary>
public class PlayerSkills : NetworkBehaviour
{
    #region Configuração
    [Header("Skill Bar")]
    [SerializeField] private SkillData[] skillBar = new SkillData[10]; // F1-F10
    
    [Header("Configurações")]
    [SerializeField] private float globalCooldown = 0.5f; // GCD entre skills
    [SerializeField] private bool allowMovementDuringCast = false;
    [SerializeField] private bool interruptOnDamage = false;
    #endregion

    #region Estado
    private SyncList<LearnedSkill> learnedSkills = new SyncList<LearnedSkill>();
    private Dictionary<string, float> cooldowns = new Dictionary<string, float>();
    private Dictionary<string, float> globalCooldowns = new Dictionary<string, float>();
    
    private bool isCasting;
    private SkillData currentCastingSkill;
    private float castStartTime;
    private Transform currentTarget;
    
    private CharacterStats stats;
    private PlayerMovement movement;
    private PlayerCombat combat;
    private Animator anim;
    #endregion

    #region Eventos
    public event System.Action<SkillData> OnSkillLearned;
    public event System.Action<SkillData> OnSkillUsed;
    public event System.Action<SkillData> OnCastStarted;
    public event System.Action<SkillData> OnCastFinished;
    public event System.Action<SkillData> OnCastInterrupted;
    public event System.Action<SkillData, float> OnCooldownStarted; // skill, remaining
    public event System.Action<int> OnSkillPointsChanged;
    #endregion

    #region Propriedades
    public bool IsCasting => isCasting;
    public SkillData CurrentCastingSkill => currentCastingSkill;
    public float CastProgress => isCasting && currentCastingSkill != null 
        ? (Time.time - castStartTime) / currentCastingSkill.castTime 
        : 0f;
    public IReadOnlyList<LearnedSkill> LearnedSkills => learnedSkills;
    
    public bool CanCast => !isCasting && !stats.IsDead && !stats.IsSilenced;
    #endregion

    void Awake()
    {
        stats = GetComponent<CharacterStats>();
        movement = GetComponent<PlayerMovement>();
        combat = GetComponent<PlayerCombat>();
        anim = GetComponent<Animator>();
    }

    void Start()
    {
        if (!isLocalPlayer)
        {
            enabled = false;
            return;
        }
    }

    void Update()
    {
        if (!isLocalPlayer) return;
        
        HandleInput();
        UpdateCooldowns();
        UpdateCasting();
    }

    #region Input
    private void HandleInput()
    {
        // F1-F10
        for (int i = 0; i < 10; i++)
        {
            if (Input.GetKeyDown(KeyCode.F1 + i))
            {
                TryUseSkill(skillBar[i]);
            }
        }
        
        // Teclas numéricas 1-0
        for (int i = 0; i < 10; i++)
        {
            KeyCode key = i == 9 ? KeyCode.Alpha0 : KeyCode.Alpha1 + i;
            if (Input.GetKeyDown(key))
            {
                TryUseSkill(skillBar[i]);
            }
        }
        
        // Tecla Esc - Cancelar cast
        if (Input.GetKeyDown(KeyCode.Escape) && isCasting)
        {
            InterruptCast();
        }
    }
    #endregion

    #region Uso de Skills
    public bool TryUseSkill(SkillData skill)
    {
        if (skill == null) return false;
        if (!CanUseSkill(skill)) return false;
        
        // Verifica target
        if (skill.requiresTarget)
        {
            if (combat.CurrentTarget == null && skill.targetType != TargetType.Self)
            {
                UIManager.Instance?.ShowErrorMessage("Selecione um alvo!");
                return false;
            }
            
            currentTarget = combat.CurrentTarget;
            
            // Verifica range
            if (currentTarget != null && skill.range > 0)
            {
                float distance = Vector3.Distance(transform.position, currentTarget.position);
                if (distance > skill.range)
                {
                    // Move até o range
                    movement.MoveToTarget(currentTarget, skill.range * 0.9f);
                    StartCoroutine(QueueSkill(skill));
                    return true;
                }
            }
        }
        
        // Inicia cast ou executa instantâneo
        if (skill.castTime > 0)
        {
            StartCast(skill);
        }
        else
        {
            ExecuteSkill(skill);
        }
        
        return true;
    }

    private bool CanUseSkill(SkillData skill)
    {
        if (skill == null) return false;
        if (stats.IsDead) return false;
        if (stats.IsSilenced) return false;
        if (isCasting) return false;
        
        // Level
        if (stats.Level < skill.requiredLevel) return false;
        
        // MP
        if (stats.CurrentMana < skill.manaCost) 
        {
            UIManager.Instance?.ShowErrorMessage("Mana insuficiente!");
            return false;
        }
        
        // Stamina (para skills físicas)
        if (skill.staminaCost > 0 && stats.CurrentStamina < skill.staminaCost)
        {
            UIManager.Instance?.ShowErrorMessage("Stamina insuficiente!");
            return false;
        }
        
        // Cooldown
        if (IsOnCooldown(skill)) 
        {
            UIManager.Instance?.ShowErrorMessage("Skill em cooldown!");
            return false;
        }
        
        // GCD
        if (IsOnGlobalCooldown(skill)) return false;
        
        // Verifica se aprendeu
        if (!HasLearnedSkill(skill.skillId)) return false;
        
        return true;
    }

    private bool IsOnCooldown(SkillData skill)
    {
        if (cooldowns.ContainsKey(skill.skillId))
        {
            return Time.time < cooldowns[skill.skillId];
        }
        return false;
    }

    private bool IsOnGlobalCooldown(SkillData skill)
    {
        if (globalCooldowns.ContainsKey(skill.skillId))
        {
            return Time.time < globalCooldowns[skill.skillId];
        }
        return false;
    }

    public float GetRemainingCooldown(SkillData skill)
    {
        if (skill == null) return 0;
        if (cooldowns.ContainsKey(skill.skillId))
        {
            return Mathf.Max(0, cooldowns[skill.skillId] - Time.time);
        }
        return 0;
    }
    #endregion

    #region Casting
    private void StartCast(SkillData skill)
    {
        isCasting = true;
        currentCastingSkill = skill;
        castStartTime = Time.time;
        
        // Para movimento se necessário
        if (!allowMovementDuringCast)
        {
            movement.SetMovementEnabled(false);
        }
        
        // Animação de cast
        anim.SetTrigger("Cast");
        anim.SetInteger("SkillId", skill.animationId);
        
        // VFX de cast
        if (skill.castVFX != null)
        {
            Instantiate(skill.castVFX, transform.position + Vector3.up, Quaternion.identity, transform);
        }
        
        OnCastStarted?.Invoke(skill);
        UIManager.Instance?.ShowCastBar(skill.skillName, skill.castTime);
    }

    private void UpdateCasting()
    {
        if (!isCasting || currentCastingSkill == null) return;
        
        float elapsed = Time.time - castStartTime;
        
        // Verifica interrupção
        if (interruptOnDamage && stats.CurrentHealth < stats.MaxHealth) // Simplificado
        {
            // Lógica de interrupção por dano
        }
        
        // Cast completo
        if (elapsed >= currentCastingSkill.castTime)
        {
            FinishCast();
        }
    }

    private void FinishCast()
    {
        if (currentCastingSkill == null) return;
        
        ExecuteSkill(currentCastingSkill);
        
        isCasting = false;
        movement.SetMovementEnabled(true);
        
        OnCastFinished?.Invoke(currentCastingSkill);
        currentCastingSkill = null;
        
        UIManager.Instance?.HideCastBar();
    }

    public void InterruptCast()
    {
        if (!isCasting) return;
        
        OnCastInterrupted?.Invoke(currentCastingSkill);
        
        isCasting = false;
        movement.SetMovementEnabled(true);
        
        // Animação de interrupção
        anim.SetTrigger("CastInterrupt");
        
        // Consome metade do MP (penalidade de interrupção)
        if (currentCastingSkill != null)
        {
            CmdConsumeResources(currentCastingSkill.skillId, 0.5f);
        }
        
        currentCastingSkill = null;
        UIManager.Instance?.HideCastBar();
    }
    #endregion

    #region Execução
    private void ExecuteSkill(SkillData skill)
    {
        // Consome recursos
        CmdConsumeResources(skill.skillId, 1f);
        
        // Aplica cooldown
        float cooldown = skill.cooldown / stats.AttackSpeed; // Attack speed reduz cooldown
        cooldowns[skill.skillId] = Time.time + cooldown;
        
        // GCD
        globalCooldowns[skill.skillId] = Time.time + globalCooldown;
        
        // Executa no servidor
        NetworkIdentity targetIdentity = currentTarget?.GetComponent<NetworkIdentity>();
        CmdExecuteSkill(skill.skillId, targetIdentity, transform.position, transform.rotation);
        
        OnSkillUsed?.Invoke(skill);
        OnCooldownStarted?.Invoke(skill, cooldown);
    }

    [Command]
    private void CmdConsumeResources(string skillId, float multiplier)
    {
        SkillData skill = SkillDatabase.Instance?.GetSkill(skillId);
        if (skill == null) return;
        
        int manaCost = Mathf.FloorToInt(skill.manaCost * multiplier);
        int staminaCost = Mathf.FloorToInt(skill.staminaCost * multiplier);
        
        stats.RestoreMana(-manaCost);
        // stats.RestoreStamina(-staminaCost); // Implementar quando stamina estiver pronto
    }

    [Command]
    private void CmdExecuteSkill(string skillId, NetworkIdentity target, Vector3 castPosition, Quaternion castRotation)
    {
        SkillData skill = SkillDatabase.Instance?.GetSkill(skillId);
        if (skill == null) return;
        
        // Anti-cheat: verifica distância
        if (target != null && skill.range > 0)
        {
            float distance = Vector3.Distance(transform.position, target.transform.position);
            if (distance > skill.range * 1.5f)
            {
                Debug.LogWarning($"[AntiCheat] Skill {skillId} de {name} em alvo muito distante!");
                return;
            }
        }
        
        // Executa efeito da skill
        skill.ExecuteServer(this, target?.GetComponent<CharacterStats>(), castPosition, castRotation);
        
        // Notifica clientes
        RpcSkillExecuted(skillId, target?.netId ?? 0, castPosition);
    }

    [ClientRpc]
    private void RpcSkillExecuted(string skillId, uint targetId, Vector3 castPosition)
    {
        SkillData skill = SkillDatabase.Instance?.GetSkill(skillId);
        if (skill == null) return;
        
        // VFX
        if (skill.vfxPrefab != null)
        {
            Vector3 spawnPos = skill.spawnAtTarget && targetId != 0 
                ? GetTargetPosition(targetId) 
                : castPosition;
            
            Instantiate(skill.vfxPrefab, spawnPos, Quaternion.identity);
        }
        
        // SFX
        if (skill.sfxClip != null)
        {
            AudioManager.Instance?.PlaySFXAt(skill.sfxClip, castPosition);
        }
        
        // Animação
        anim.SetTrigger("Skill" + skill.animationId);
        
        // Shake de câmera para skills impactantes
        if (skill.cameraShake > 0 && isLocalPlayer)
        {
            CameraShake.Instance?.Shake(skill.cameraShake, skill.cameraShakeDuration);
        }
    }

    private Vector3 GetTargetPosition(uint targetId)
    {
        if (NetworkIdentity.spawned.TryGetValue(targetId, out NetworkIdentity identity))
        {
            return identity.transform.position;
        }
        return transform.position;
    }
    #endregion

    #region Aprendizado de Skills
    [Server]
    public bool LearnSkill(SkillData skill)
    {
        if (skill == null) return false;
        if (HasLearnedSkill(skill.skillId)) return false;
        if (stats.SkillPoints < skill.skillPointCost) return false;
        
        // Verifica requisitos
        foreach (string reqId in skill.requiredSkills)
        {
            if (!HasLearnedSkill(reqId)) return false;
        }
        
        learnedSkills.Add(new LearnedSkill
        {
            skillId = skill.skillId,
            level = 1,
            experience = 0
        });
        
        // Consome pontos
        // stats.SkillPoints -= skill.skillPointCost; // Implementar
        
        OnSkillLearned?.Invoke(skill);
        return true;
    }

    [Server]
    public bool UpgradeSkill(string skillId)
    {
        int index = GetLearnedSkillIndex(skillId);
        if (index < 0) return false;
        
        var learned = learnedSkills[index];
        SkillData skill = SkillDatabase.Instance?.GetSkill(skillId);
        
        if (skill == null || learned.level >= skill.maxLevel) return false;
        if (stats.SkillPoints < skill.upgradeCost) return false;
        
        learned.level++;
        learnedSkills[index] = learned;
        
        // stats.SkillPoints -= skill.upgradeCost;
        
        return true;
    }

    public bool HasLearnedSkill(string skillId)
    {
        foreach (var learned in learnedSkills)
        {
            if (learned.skillId == skillId) return true;
        }
        return false;
    }

    public int GetSkillLevel(string skillId)
    {
        foreach (var learned in learnedSkills)
        {
            if (learned.skillId == skillId) return learned.level;
        }
        return 0;
    }

    private int GetLearnedSkillIndex(string skillId)
    {
        for (int i = 0; i < learnedSkills.Count; i++)
        {
            if (learnedSkills[i].skillId == skillId) return i;
        }
        return -1;
    }
    #endregion

    #region Skill Bar
    public void AssignSkillToSlot(SkillData skill, int slot)
    {
        if (slot < 0 || slot >= skillBar.Length) return;
        skillBar[slot] = skill;
    }

    public SkillData GetSkillInSlot(int slot)
    {
        if (slot < 0 || slot >= skillBar.Length) return null;
        return skillBar[slot];
    }
    #endregion

    #region Cooldowns
    private void UpdateCooldowns()
    {
        // Remove cooldowns expirados
        List<string> toRemove = new List<string>();
        foreach (var kvp in cooldowns)
        {
            if (Time.time >= kvp.Value)
            {
                toRemove.Add(kvp.Key);
            }
        }
        foreach (var key in toRemove)
        {
            cooldowns.Remove(key);
        }
        
        toRemove.Clear();
        foreach (var kvp in globalCooldowns)
        {
            if (Time.time >= kvp.Value)
            {
                toRemove.Add(kvp.Key);
            }
        }
        foreach (var key in toRemove)
        {
            globalCooldowns.Remove(key);
        }
    }
    #endregion

    #region Queue
    private System.Collections.IEnumerator QueueSkill(SkillData skill)
    {
        yield return new WaitUntil(() => 
            currentTarget == null || 
            Vector3.Distance(transform.position, currentTarget.position) <= skill.range ||
            !movement.IsMoving);
        
        if (currentTarget != null && Vector3.Distance(transform.position, currentTarget.position) <= skill.range)
        {
            TryUseSkill(skill);
        }
    }
    #endregion
}

#region Structs e Enums
[System.Serializable]
public struct LearnedSkill : Mirror.NetworkMessage
{
    public string skillId;
    public int level;
    public int experience; // Para skills que upam com uso
}

public enum TargetType
{
    Self,
    Single,
    Area,
    Cone,
    Line,
    Party,
    Guild
}

public enum SkillType
{
    Physical,
    Magical,
    Buff,
    Debuff,
    Heal,
    Summon,
    Transform,
    Passive
}
#endregion
'''

with open(f"{base_path}/Scripts/Skills/PlayerSkills.cs", "w") as f:
    f.write(player_skills)

print("✅ PlayerSkills.cs criado!")
