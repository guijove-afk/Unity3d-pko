# ItemDatabase.cs
item_database = '''using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// Database central de itens do Tales of Pirates
/// Carrega todos os ScriptableObjects de itens
/// </summary>
public class ItemDatabase : MonoBehaviour
{
    public static ItemDatabase Instance { get; private set; }
    
    [Header("Itens")]
    [SerializeField] private List<ItemData> allItems = new List<ItemData>();
    
    private Dictionary<string, ItemData> itemLookup = new Dictionary<string, ItemData>();
    private Dictionary<string, EquipmentData> equipmentLookup = new Dictionary<string, EquipmentData>();
    private Dictionary<int, GemData> gemLookup = new Dictionary<int, GemData>();
    private Dictionary<string, SetData> setLookup = new Dictionary<string, SetData>();
    
    void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
            BuildDatabase();
        }
        else
        {
            Destroy(gameObject);
        }
    }
    
    private void BuildDatabase()
    {
        itemLookup.Clear();
        equipmentLookup.Clear();
        gemLookup.Clear();
        setLookup.Clear();
        
        foreach (var item in allItems)
        {
            if (item == null) continue;
            
            itemLookup[item.itemId] = item;
            
            if (item is EquipmentData equip)
            {
                equipmentLookup[item.itemId] = equip;
            }
            
            if (item is GemData gem)
            {
                gemLookup[gem.GetInstanceID()] = gem;
            }
        }
        
        // Carrega sets
        SetData[] sets = Resources.LoadAll<SetData>("Sets");
        foreach (var set in sets)
        {
            if (set != null)
            {
                setLookup[set.setName] = set;
            }
        }
    }
    
    public ItemData GetItem(string itemId)
    {
        if (itemLookup.TryGetValue(itemId, out ItemData item))
        {
            return item;
        }
        return null;
    }
    
    public EquipmentData GetEquipment(string itemId)
    {
        if (equipmentLookup.TryGetValue(itemId, out EquipmentData equip))
        {
            return equip;
        }
        return null;
    }
    
    public GemData GetGem(int gemId)
    {
        if (gemLookup.TryGetValue(gemId, out GemData gem))
        {
            return gem;
        }
        return null;
    }
    
    public SetData GetSetData(string setName)
    {
        if (setLookup.TryGetValue(setName, out SetData set))
        {
            return set;
        }
        return null;
    }
    
    public List<ItemData> GetItemsByType(ItemType type)
    {
        List<ItemData> result = new List<ItemData>();
        foreach (var item in itemLookup.Values)
        {
            if (item.itemType == type)
            {
                result.Add(item);
            }
        }
        return result;
    }
    
    public List<ItemData> GetItemsByClass(CharacterClass characterClass)
    {
        List<ItemData> result = new List<ItemData>();
        foreach (var item in itemLookup.Values)
        {
            if (item is EquipmentData equip && equip.requiredClass == characterClass)
            {
                result.Add(item);
            }
        }
        return result;
    }
    
    public void RegisterItem(ItemData item)
    {
        if (item == null || string.IsNullOrEmpty(item.itemId)) return;
        
        if (!itemLookup.ContainsKey(item.itemId))
        {
            allItems.Add(item);
            itemLookup[item.itemId] = item;
            
            if (item is EquipmentData equip)
            {
                equipmentLookup[item.itemId] = equip;
            }
        }
    }
}
'''

# SkillDatabase.cs
skill_database = '''using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// Database central de skills
/// </summary>
public class SkillDatabase : MonoBehaviour
{
    public static SkillDatabase Instance { get; private set; }
    
    [Header("Skills")]
    [SerializeField] private List<SkillData> allSkills = new List<SkillData>();
    
    private Dictionary<string, SkillData> skillLookup = new Dictionary<string, SkillData>();
    
    void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
            BuildDatabase();
        }
        else
        {
            Destroy(gameObject);
        }
    }
    
    private void BuildDatabase()
    {
        skillLookup.Clear();
        
        foreach (var skill in allSkills)
        {
            if (skill == null) continue;
            skillLookup[skill.skillId] = skill;
        }
    }
    
    public SkillData GetSkill(string skillId)
    {
        if (skillLookup.TryGetValue(skillId, out SkillData skill))
        {
            return skill;
        }
        return null;
    }
    
    public List<SkillData> GetSkillsByClass(CharacterClass characterClass)
    {
        List<SkillData> result = new List<SkillData>();
        foreach (var skill in skillLookup.Values)
        {
            if (skill.requiredClass == characterClass)
            {
                result.Add(skill);
            }
        }
        return result;
    }
    
    public List<SkillData> GetSkillsByLevel(int level)
    {
        List<SkillData> result = new List<SkillData>();
        foreach (var skill in skillLookup.Values)
        {
            if (skill.requiredLevel <= level)
            {
                result.Add(skill);
            }
        }
        return result;
    }
    
    public void RegisterSkill(SkillData skill)
    {
        if (skill == null || string.IsNullOrEmpty(skill.skillId)) return;
        
        if (!skillLookup.ContainsKey(skill.skillId))
        {
            allSkills.Add(skill);
            skillLookup[skill.skillId] = skill;
        }
    }
}
'''

# ClassDatabase.cs
class_database = '''using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// Database de classes do Tales of Pirates
/// </summary>
public class ClassDatabase : MonoBehaviour
{
    public static ClassDatabase Instance { get; private set; }
    
    [Header("Classes")]
    [SerializeField] private List<ClassData> allClasses = new List<ClassData>();
    
    private Dictionary<CharacterClass, ClassData> classLookup = new Dictionary<CharacterClass, ClassData>();
    
    void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
            BuildDatabase();
        }
        else
        {
            Destroy(gameObject);
        }
    }
    
    private void BuildDatabase()
    {
        classLookup.Clear();
        
        foreach (var classData in allClasses)
        {
            if (classData == null) continue;
            classLookup[classData.characterClass] = classData;
        }
    }
    
    public ClassData GetClassData(CharacterClass characterClass)
    {
        if (classLookup.TryGetValue(characterClass, out ClassData data))
        {
            return data;
        }
        return null;
    }
    
    public List<ClassData> GetStartingClasses()
    {
        List<ClassData> result = new List<ClassData>();
        foreach (var classData in classLookup.Values)
        {
            // Classes iniciais: Swordsman, Hunter, Explorer, Herbalist
            if ((int)classData.characterClass <= 4)
            {
                result.Add(classData);
            }
        }
        return result;
    }
    
    public List<ClassData> GetEvolutionClasses(CharacterClass currentClass)
    {
        List<ClassData> result = new List<ClassData>();
        foreach (var classData in classLookup.Values)
        {
            if (classData.characterClass > currentClass && 
                (int)classData.characterClass / 10 == (int)currentClass / 10)
            {
                result.Add(classData);
            }
        }
        return result;
    }
}
'''

# DamagePopupManager.cs
damage_popup = '''using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// Gerenciador de números flutuantes de dano/cura
/// </summary>
public class DamagePopupManager : MonoBehaviour
{
    public static DamagePopupManager Instance { get; private set; }
    
    [Header("Configurações")]
    [SerializeField] private GameObject damagePopupPrefab;
    [SerializeField] private int poolSize = 50;
    [SerializeField] private float popupDuration = 1.5f;
    [SerializeField] private float popupSpeed = 2f;
    
    [Header("Cores")]
    [SerializeField] private Color normalDamageColor = Color.white;
    [SerializeField] private Color criticalDamageColor = Color.red;
    [SerializeField] private Color healColor = Color.green;
    [SerializeField] private Color missColor = Color.gray;
    
    private Queue<GameObject> popupPool = new Queue<GameObject>();
    private List<DamagePopup> activePopups = new List<DamagePopup>();
    
    void Awake()
    {
        if (Instance == null) Instance = this;
        else Destroy(gameObject);
    }
    
    void Start()
    {
        InitializePool();
    }
    
    private void InitializePool()
    {
        for (int i = 0; i < poolSize; i++)
        {
            GameObject popup = Instantiate(damagePopupPrefab, transform);
            popup.SetActive(false);
            popupPool.Enqueue(popup);
        }
    }
    
    public void ShowDamage(Vector3 position, int damage, bool isCritical = false)
    {
        GameObject popup = GetPopup();
        if (popup == null) return;
        
        popup.transform.position = position;
        popup.SetActive(true);
        
        DamagePopup dp = popup.GetComponent<DamagePopup>();
        if (dp != null)
        {
            dp.Initialize(damage.ToString(), isCritical ? criticalDamageColor : normalDamageColor, 
                isCritical ? 1.5f : 1f, popupDuration);
        }
    }
    
    public void ShowHeal(Vector3 position, int amount)
    {
        GameObject popup = GetPopup();
        if (popup == null) return;
        
        popup.transform.position = position;
        popup.SetActive(true);
        
        DamagePopup dp = popup.GetComponent<DamagePopup>();
        if (dp != null)
        {
            dp.Initialize("+" + amount, healColor, 1f, popupDuration);
        }
    }
    
    public void ShowMiss(Vector3 position)
    {
        GameObject popup = GetPopup();
        if (popup == null) return;
        
        popup.transform.position = position;
        popup.SetActive(true);
        
        DamagePopup dp = popup.GetComponent<DamagePopup>();
        if (dp != null)
        {
            dp.Initialize("MISS", missColor, 1f, popupDuration);
        }
    }
    
    private GameObject GetPopup()
    {
        if (popupPool.Count > 0)
        {
            return popupPool.Dequeue();
        }
        return null;
    }
    
    public void ReturnPopup(GameObject popup)
    {
        popup.SetActive(false);
        popupPool.Enqueue(popup);
    }
}

public class DamagePopup : MonoBehaviour
{
    private TextMesh textMesh;
    private float duration;
    private float elapsed;
    private Vector3 startPosition;
    private Color color;
    
    void Awake()
    {
        textMesh = GetComponent<TextMesh>();
    }
    
    public void Initialize(string text, Color color, float scale, float duration)
    {
        this.duration = duration;
        this.elapsed = 0;
        this.startPosition = transform.position;
        this.color = color;
        
        if (textMesh != null)
        {
            textMesh.text = text;
            textMesh.color = color;
            transform.localScale = Vector3.one * scale;
        }
    }
    
    void Update()
    {
        elapsed += Time.deltaTime;
        
        // Move para cima
        transform.position = startPosition + Vector3.up * (elapsed * 2f);
        
        // Fade out
        float alpha = 1 - (elapsed / duration);
        if (textMesh != null)
        {
            textMesh.color = new Color(color.r, color.g, color.b, alpha);
        }
        
        if (elapsed >= duration)
        {
            DamagePopupManager.Instance?.ReturnPopup(gameObject);
        }
    }
}
'''

# AudioManager.cs
audio_manager = '''using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// Gerenciador de áudio do Tales of Pirates
/// </summary>
public class AudioManager : MonoBehaviour
{
    public static AudioManager Instance { get; private set; }
    
    [Header("Configurações")]
    [SerializeField] private int sfxSourceCount = 10;
    [SerializeField] private AudioSource musicSource;
    
    private List<AudioSource> sfxSources = new List<AudioSource>();
    private int currentSfxIndex = 0;
    
    [Header("Volumes")]
    [SerializeField] [Range(0, 1)] private float masterVolume = 1f;
    [SerializeField] [Range(0, 1)] private float musicVolume = 0.7f;
    [SerializeField] [Range(0, 1)] private float sfxVolume = 0.8f;
    
    void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
            InitializeSources();
        }
        else
        {
            Destroy(gameObject);
        }
    }
    
    private void InitializeSources()
    {
        for (int i = 0; i < sfxSourceCount; i++)
        {
            AudioSource source = gameObject.AddComponent<AudioSource>();
            source.playOnAwake = false;
            sfxSources.Add(source);
        }
    }
    
    public void PlaySFX(AudioClip clip)
    {
        if (clip == null) return;
        
        AudioSource source = GetNextSfxSource();
        source.clip = clip;
        source.volume = sfxVolume * masterVolume;
        source.Play();
    }
    
    public void PlaySFXAt(AudioClip clip, Vector3 position)
    {
        if (clip == null) return;
        
        // Calcula volume baseado na distância (simplificado)
        float distance = Vector3.Distance(Camera.main.transform.position, position);
        float volume = Mathf.Clamp01(1 - (distance / 50f)); // 50m de range
        
        AudioSource source = GetNextSfxSource();
        source.clip = clip;
        source.volume = volume * sfxVolume * masterVolume;
        source.spatialBlend = 1f; // 3D
        source.transform.position = position;
        source.Play();
    }
    
    public void PlayMusic(AudioClip clip, bool loop = true)
    {
        if (musicSource == null || clip == null) return;
        
        musicSource.clip = clip;
        musicSource.loop = loop;
        musicSource.volume = musicVolume * masterVolume;
        musicSource.Play();
    }
    
    public void StopMusic()
    {
        if (musicSource != null)
        {
            musicSource.Stop();
        }
    }
    
    private AudioSource GetNextSfxSource()
    {
        AudioSource source = sfxSources[currentSfxIndex];
        currentSfxIndex = (currentSfxIndex + 1) % sfxSourceCount;
        return source;
    }
    
    public void SetMasterVolume(float volume)
    {
        masterVolume = Mathf.Clamp01(volume);
        UpdateVolumes();
    }
    
    public void SetMusicVolume(float volume)
    {
        musicVolume = Mathf.Clamp01(volume);
        UpdateVolumes();
    }
    
    public void SetSFXVolume(float volume)
    {
        sfxVolume = Mathf.Clamp01(volume);
        UpdateVolumes();
    }
    
    private void UpdateVolumes()
    {
        if (musicSource != null)
        {
            musicSource.volume = musicVolume * masterVolume;
        }
    }
}
'''
