
# Script 3: PlayerCombat.cs - Sistema de combate completo fiel ao TOP

player_combat = '''using UnityEngine;
using Mirror;
using System.Collections;
using System.Collections.Generic;

/// <summary>
/// Sistema de combate completo baseado em Tales of Pirates
/// Suporta: ataque normal, skills, combo, target lock, auto-attack
/// </summary>
[RequireComponent(typeof(PlayerMovement))]
[RequireComponent(typeof(CharacterStats))]
[RequireComponent(typeof(Animator))]
public class PlayerCombat : NetworkBehaviour
{
    #region Configuração
    [Header("Ataque Básico")]
    [SerializeField] private float attackRange = 2.5f;
    [SerializeField] private float attackCooldown = 0.8f;
    [SerializeField] private float attackCastTime = 0.35f; // Momento do impacto na animação
    [SerializeField] private int baseDamage = 10;
    
    [Header("Combo")]
    [SerializeField] private int maxCombo = 3;
    [SerializeField] private float comboResetTime = 2f;
    [SerializeField] private float comboDamageMultiplier = 0.15f;
    
    [Header("Layers")]
    [SerializeField] private LayerMask enemyLayers;
    [SerializeField] private LayerMask attackableLayers;
    
    [Header("Animação")]
    [SerializeField] private string attackTrigger = "Attack";
    [SerializeField] private string comboParam = "ComboCount";
    [SerializeField] private string attackTypeParam = "AttackType";
    
    [Header("VFX/SFX")]
    [SerializeField] private ParticleSystem attackVFX;
    [SerializeField] private ParticleSystem criticalVFX;
    [SerializeField] private AudioClip attackSFX;
    [SerializeField] private AudioClip criticalSFX;
    [SerializeField] private AudioClip missSFX;
    
    [Header("Configurações TOP")]
    [SerializeField] private bool autoAttack = false; // Auto-attack contínuo
    [SerializeField] private float targetLockRange = 15f;
    [SerializeField] private bool faceTargetOnAttack = true;
    #endregion

    #region Componentes
    private PlayerMovement movement;
    private CharacterStats stats;
    private Animator anim;
    private AudioSource audioSource;
    #endregion

    #region Estado
    private float lastAttackTime;
    private bool isAttacking;
    private Transform currentTarget;
    private int currentCombo;
    private float lastComboTime;
    private bool isAutoAttacking;
    private List<Transform> nearbyEnemies = new List<Transform>();
    #endregion

    #region Eventos
    public event System.Action<Transform, int, bool> OnAttackLanded; // target, damage, isCritical
    public event System.Action OnAttackStarted;
    public event System.Action OnAttackFinished;
    public event System.Action<Transform> OnTargetChanged;
    public event System.Action OnComboReset;
    #endregion

    #region Propriedades
    public bool CanAttack => !isAttacking && !stats.IsDead && !stats.IsStunned && Time.time >= lastAttackTime + GetCurrentCooldown();
    public bool IsAttacking => isAttacking;
    public Transform CurrentTarget => currentTarget;
    public float AttackRange => attackRange;
    public int CurrentCombo => currentCombo;
    public bool IsAutoAttacking => isAutoAttacking;
    #endregion

    void Awake()
    {
        movement = GetComponent<PlayerMovement>();
        stats = GetComponent<CharacterStats>();
        anim = GetComponent<Animator>();
        audioSource = GetComponent<AudioSource>();
        
        if (audioSource == null)
        {
            audioSource = gameObject.AddComponent<AudioSource>();
        }
    }

    void Start()
    {
        if (!isLocalPlayer)
        {
            enabled = false;
            return;
        }
    }

    void Update()
    {
        if (!isLocalPlayer) return;
        
        HandleInput();
        UpdateTarget();
        UpdateCombo();
        
        // Auto-attack
        if (isAutoAttacking && currentTarget != null && CanAttack)
        {
            TryAttackTarget(currentTarget);
        }
    }

    #region Input
    private void HandleInput()
    {
        // Tecla Space - Atacar alvo atual ou buscar novo
        if (Input.GetKeyDown(KeyCode.Space))
        {
            if (currentTarget != null)
            {
                TryAttackTarget(currentTarget);
            }
            else
            {
                TryAttackNearest();
            }
        }
        
        // Tecla Tab - Trocar alvo
        if (Input.GetKeyDown(KeyCode.Tab))
        {
            CycleTarget();
        }
        
        // Ctrl + Click - Auto-attack toggle
        if (Input.GetKey(KeyCode.LeftControl) && Input.GetMouseButtonDown(0))
        {
            ToggleAutoAttack();
        }
        
        // Tecla Z - Cancelar ataque/alvo
        if (Input.GetKeyDown(KeyCode.Z))
        {
            ClearTarget();
        }
    }
    #endregion

    #region Ataque
    public void TryAttackNearest()
    {
        UpdateNearbyEnemies();
        
        if (nearbyEnemies.Count == 0) return;
        
        Transform nearest = GetBestTarget(nearbyEnemies);
        if (nearest != null)
        {
            SetTarget(nearest);
            TryAttackTarget(nearest);
        }
    }

    public void TryAttackTarget(Transform target)
    {
        if (!CanAttack || target == null) return;
        
        float distance = Vector3.Distance(transform.position, target.position);
        
        if (distance > attackRange)
        {
            // Move até o alvo
            movement.MoveToTarget(target, attackRange * 0.8f);
            
            // Queue attack para quando chegar
            StartCoroutine(QueueAttack(target));
            return;
        }
        
        // Executa ataque
        StartCoroutine(AttackSequence(target));
    }

    private IEnumerator AttackSequence(Transform target)
    {
        isAttacking = true;
        lastAttackTime = Time.time;
        
        // Para movimento
        movement.SetMovementEnabled(false);
        
        // Combo system
        currentCombo++;
        if (currentCombo > maxCombo)
        {
            currentCombo = 1;
        }
        lastComboTime = Time.time;
        
        // Animação
        anim.SetInteger(comboParam, currentCombo);
        anim.SetTrigger(attackTrigger);
        
        OnAttackStarted?.Invoke();
        
        // Face target
        if (faceTargetOnAttack && target != null)
        {
            movement.LookAt(target.position);
        }
        
        // Aguarda cast time (momento do impacto)
        yield return new WaitForSeconds(attackCastTime);
        
        // Verifica se ainda válido
        if (target != null && Vector3.Distance(transform.position, target.position) <= attackRange * 1.2f)
        {
            // Executa ataque no servidor
            NetworkIdentity targetIdentity = target.GetComponent<NetworkIdentity>();
            if (targetIdentity != null)
            {
                int damage = CalculateDamage();
                bool isCritical = CheckCritical();
                
                CmdPerformAttack(targetIdentity, damage, isCritical, currentCombo);
            }
        }
        
        // Aguarda fim do cooldown
        float remainingCooldown = GetCurrentCooldown() - attackCastTime;
        if (remainingCooldown > 0)
        {
            yield return new WaitForSeconds(remainingCooldown);
        }
        
        isAttacking = false;
        movement.SetMovementEnabled(true);
        OnAttackFinished?.Invoke();
    }

    private IEnumerator QueueAttack(Transform target)
    {
        yield return new WaitUntil(() => 
            target == null || 
            Vector3.Distance(transform.position, target.position) <= attackRange ||
            !movement.IsMoving);
        
        if (target != null && Vector3.Distance(transform.position, target.position) <= attackRange)
        {
            TryAttackTarget(target);
        }
    }
    #endregion

    #region Cálculos de Dano
    private int CalculateDamage()
    {
        // Fórmula TOP: ATK + STR*0.5 + Level - DEF alvo
        int damage = stats.Attack;
        damage += Mathf.FloorToInt(stats.Strength * 0.5f);
        damage += stats.Level;
        
        // Combo multiplier
        if (currentCombo > 1)
        {
            damage += Mathf.FloorToInt(damage * comboDamageMultiplier * (currentCombo - 1));
        }
        
        // Randomização
        float randomFactor = Random.Range(0.9f, 1.1f);
        damage = Mathf.FloorToInt(damage * randomFactor);
        
        return Mathf.Max(1, damage);
    }

    private bool CheckCritical()
    {
        float critChance = stats.CriticalRate;
        return Random.Range(0f, 100f) < critChance;
    }

    private float GetCurrentCooldown()
    {
        // Attack speed afeta cooldown
        return attackCooldown / stats.AttackSpeed;
    }
    #endregion

    #region Targeting
    public void SetTarget(Transform target)
    {
        if (currentTarget != target)
        {
            currentTarget = target;
            OnTargetChanged?.Invoke(target);
            
            // Mostra target indicator
            TargetIndicatorManager.Instance?.SetTarget(target);
        }
    }

    public void ClearTarget()
    {
        currentTarget = null;
        isAutoAttacking = false;
        OnTargetChanged?.Invoke(null);
        TargetIndicatorManager.Instance?.ClearTarget();
    }

    private void UpdateTarget()
    {
        if (currentTarget == null) return;
        
        // Verifica se alvo ainda válido
        CharacterStats targetStats = currentTarget.GetComponent<CharacterStats>();
        
        if (targetStats != null && targetStats.IsDead)
        {
            // Alvo morreu, busca próximo
            if (isAutoAttacking)
            {
                TryAttackNearest();
            }
            else
            {
                ClearTarget();
            }
            return;
        }
        
        // Verifica se saiu de range
        float distance = Vector3.Distance(transform.position, currentTarget.position);
        if (distance > targetLockRange)
        {
            ClearTarget();
        }
    }

    private void CycleTarget()
    {
        UpdateNearbyEnemies();
        
        if (nearbyEnemies.Count == 0)
        {
            ClearTarget();
            return;
        }
        
        int currentIndex = nearbyEnemies.IndexOf(currentTarget);
        int nextIndex = (currentIndex + 1) % nearbyEnemies.Count;
        
        SetTarget(nearbyEnemies[nextIndex]);
    }

    private void UpdateNearbyEnemies()
    {
        nearbyEnemies.Clear();
        
        Collider[] hits = Physics.OverlapSphere(transform.position, targetLockRange, enemyLayers);
        foreach (var hit in hits)
        {
            if (hit.transform != transform)
            {
                CharacterStats cs = hit.GetComponent<CharacterStats>();
                if (cs != null && !cs.IsDead)
                {
                    nearbyEnemies.Add(hit.transform);
                }
            }
        }
        
        // Ordena por distância
        nearbyEnemies.Sort((a, b) => 
            Vector3.Distance(transform.position, a.position).CompareTo(
            Vector3.Distance(transform.position, b.position)));
    }

    private Transform GetBestTarget(List<Transform> enemies)
    {
        if (enemies.Count == 0) return null;
        
        Transform best = null;
        float bestScore = float.MaxValue;
        
        foreach (var enemy in enemies)
        {
            float distance = Vector3.Distance(transform.position, enemy.position);
            float angle = Vector3.Angle(transform.forward, enemy.position - transform.position);
            
            // Score: prioriza distância e ângulo frontal
            float score = distance + angle * 0.05f;
            
            if (score < bestScore)
            {
                bestScore = score;
                best = enemy;
            }
        }
        
        return best;
    }
    #endregion

    #region Combo System
    private void UpdateCombo()
    {
        if (currentCombo > 0 && Time.time > lastComboTime + comboResetTime)
        {
            currentCombo = 0;
            OnComboReset?.Invoke();
        }
    }
    #endregion

    #region Auto Attack
    private void ToggleAutoAttack()
    {
        isAutoAttacking = !isAutoAttacking;
        
        if (isAutoAttacking && currentTarget == null)
        {
            TryAttackNearest();
        }
        
        UIManager.Instance?.ShowAutoAttackIndicator(isAutoAttacking);
    }
    #endregion

    #region Comandos de Rede
    [Command]
    private void CmdPerformAttack(NetworkIdentity targetIdentity, int damage, bool isCritical, int combo)
    {
        if (targetIdentity == null) return;
        
        CharacterStats targetStats = targetIdentity.GetComponent<CharacterStats>();
        if (targetStats == null || targetStats.IsDead) return;
        
        // Verificação anti-cheat: distância
        float distance = Vector3.Distance(transform.position, targetIdentity.transform.position);
        if (distance > attackRange * 1.5f)
        {
            Debug.LogWarning($"[AntiCheat] Ataque de {name} em {targetIdentity.name} muito distante: {distance}");
            return;
        }
        
        // Aplica dano
        targetStats.TakeDamage(damage, netId, DamageType.Physical);
        
        // Notifica clientes
        RpcOnAttackLanded(targetIdentity.netId, damage, isCritical, combo);
    }

    [ClientRpc]
    private void RpcOnAttackLanded(uint targetId, int damage, bool isCritical, int combo)
    {
        // Encontra alvo
        if (NetworkIdentity.spawned.TryGetValue(targetId, out NetworkIdentity targetIdentity))
        {
            Transform target = targetIdentity.transform;
            
            // VFX
            if (isCritical && criticalVFX != null)
            {
                Instantiate(criticalVFX, target.position + Vector3.up, Quaternion.identity);
                AudioManager.Instance?.PlaySFXAt(criticalSFX, target.position);
            }
            else if (attackVFX != null)
            {
                Instantiate(attackVFX, target.position + Vector3.up * 0.5f, Quaternion.identity);
            }
            
            // SFX
            if (!isCritical && attackSFX != null)
            {
                AudioManager.Instance?.PlaySFXAt(attackSFX, target.position);
            }
            
            // Damage popup
            DamagePopupManager.Instance?.ShowDamage(target.position + Vector3.up * 2f, damage, isCritical);
            
            // Evento local
            OnAttackLanded?.Invoke(target, damage, isCritical);
        }
    }
    #endregion

    #region Utilidades
    public void EnableCombat(bool enabled)
    {
        this.enabled = enabled;
        if (!enabled)
        {
            ClearTarget();
            isAttacking = false;
        }
    }

    void OnDrawGizmosSelected()
    {
        Gizmos.color = Color.red;
        Gizmos.DrawWireSphere(transform.position, attackRange);
        
        Gizmos.color = Color.yellow;
        Gizmos.DrawWireSphere(transform.position, targetLockRange);
        
        if (currentTarget != null)
        {
            Gizmos.color = Color.red;
            Gizmos.DrawLine(transform.position, currentTarget.position);
        }
    }
    #endregion
}

/// <summary>
/// Singleton para indicador de alvo (implementação básica)
/// </summary>
public class TargetIndicatorManager : MonoBehaviour
{
    public static TargetIndicatorManager Instance { get; private set; }
    
    [SerializeField] private GameObject targetIndicatorPrefab;
    private GameObject currentIndicator;
    
    void Awake()
    {
        if (Instance == null) Instance = this;
        else Destroy(gameObject);
    }
    
    public void SetTarget(Transform target)
    {
        if (currentIndicator != null) Destroy(currentIndicator);
        if (target == null) return;
        
        currentIndicator = Instantiate(targetIndicatorPrefab, target);
        currentIndicator.transform.localPosition = Vector3.zero;
    }
    
    public void ClearTarget()
    {
        if (currentIndicator != null) Destroy(currentIndicator);
    }
}
'''
